124 - (instancetype)initWithMusicSequence:(MusicSequence)musicSequence error:(NSError **)error
125 {
126  error = error ? error : &(NSError *__autoreleasing){ nil };
127 
128  if (self = [super init]) {
129  OSStatus err = MusicSequenceSetUserCallback(musicSequence, MIKSequenceCallback, (__bridge void *)self);
130  if (err) {
131  NSLog(@"MusicSequenceSetUserCallback() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
132  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
133  return nil;
134  }
135  self.musicSequence = musicSequence;
136 
137  MusicTrack tempoTrack;
138  err = MusicSequenceGetTempoTrack(musicSequence, &tempoTrack);
139  if (err) {
140  NSLog(@"MusicSequenceGetTempoTrack() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
141  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
142  return nil;
143  }
144  self.tempoTrack = [MIKMIDITrack trackWithSequence:self musicTrack:tempoTrack];
145 
146  UInt32 numTracks = 0;
147  err = MusicSequenceGetTrackCount(musicSequence, &numTracks);
148  if (err) {
149  NSLog(@"MusicSequenceGetTrackCount() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
150  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
151  return nil;
152  }
153  NSMutableArray *tracks = [NSMutableArray arrayWithCapacity:numTracks];
154 
155  for (UInt32 i = 0; i < numTracks; i++) {
156  MusicTrack musicTrack;
157  err = MusicSequenceGetIndTrack(musicSequence, i, &musicTrack);
158  if (err){
159  NSLog(@"MusicSequenceGetIndTrack() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
160  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
161  return nil;
162  }
163  [tracks addObject:[MIKMIDITrack trackWithSequence:self musicTrack:musicTrack]];
164  }
165  self.internalTracks = tracks;
166  self.length = MIKMIDISequenceLongestTrackLength;
167  }
168 
169  return self;
170 }

258 static void MIKSequenceCallback(void *inClientData, MusicSequence inSequence, MusicTrack inTrack, MusicTimeStamp inEventTime, const MusicEventUserData *inEventData, MusicTimeStamp inStartSliceBeat, MusicTimeStamp inEndSliceBeat)
259 {
260  MIKMIDISequence *self = (__bridge MIKMIDISequence *)inClientData;
261  if (!self.callBackBlock) return;
262 
263  UInt32 trackIndex;
264  OSStatus err = MusicSequenceGetTrackIndex(inSequence, inTrack, &trackIndex);
265  if (err) {
266  NSLog(@"MusicSequenceGetTrackIndex() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
267  return;
268  }
269 
270  MIKMIDITrack *track = self.tracks[trackIndex];
271  if (track && self.callBackBlock) {
272  self.callBackBlock(track, inEventTime, inEventData, inStartSliceBeat, inEndSliceBeat);
273  }
274 }

146 - (NSString *)createNewConnectionToken
147 {
148  __block NSString *uuidString = nil;
149  dispatch_sync(self.handlerTokenQueue, ^{
150  do { // Very unlikely, but just to be safe
151  CFUUIDRef uuid = CFUUIDCreate(kCFAllocatorDefault);
152  uuidString = CFBridgingRelease(CFUUIDCreateString(kCFAllocatorDefault, uuid));
153  CFRelease(uuid);
154  MIKMIDIConnectionTokenAndEventHandler *existingPair = nil;
155  for (NSArray *handlerPairs in self.handlerTokenPairsByEndpoint.objectEnumerator) {
156  for (MIKMIDIConnectionTokenAndEventHandler *pair in handlerPairs) {
157  if ([pair.connectionToken isEqualToString:uuidString]) {
158  existingPair = pair;
159  break;
160  }
161  }
162  }
163  if (!existingPair) break;
164  } while (1);
165  });
166  return uuidString;
167 }

214 - (BOOL)commandIsPossibleMSBOf14BitCommand:(MIKMIDICommand *)command
215 {
216  if (command.commandType != MIKMIDICommandTypeControlChange) return NO;

224 - (NSArray *)commandsByCoalescingCommands:(NSArray *)commands
225 {
226  NSMutableArray *coalescedCommands = [commands mutableCopy];
227  MIKMIDICommand *lastCommand = commands.firstObject;
228  for (MIKMIDICommand *command in commands) {
229  MIKMIDIControlChangeCommand *coalesced =
230  [MIKMIDIControlChangeCommand commandByCoalescingMSBCommand:(MIKMIDIControlChangeCommand *)lastCommand
231  andLSBCommand:(MIKMIDIControlChangeCommand *)command];
232  if (coalesced) {
233  [coalescedCommands removeObject:command];
234  NSUInteger lastCommandIndex = [coalescedCommands indexOfObject:lastCommand];
235  [coalescedCommands replaceObjectAtIndex:lastCommandIndex withObject:coalesced];
236  }
237  lastCommand = command;
238  }
239  return [coalescedCommands copy];
240 }

242 - (BOOL)coalesceSysexFromMIDIPacket:(const MIDIPacket *)packet toCommandInArray:(NSMutableArray **)commandsArray
243 {
244  const Byte *data = packet->data;
245 
246  Byte firstByte = data[0];
247 
248  if (self.sysexData == nil) {
249  // Check for Sysex Begin
250  if (firstByte != kMIKMIDISysexBeginDelimiter) {
251  return NO;
252  }
253 
254  self.sysexData = [NSMutableData new];
255  self.sysexStartTimeStamp = packet->timeStamp;
256  } else if (firstByte > 0x7F && firstByte != kMIKMIDISysexEndDelimiter) {
257  // Invalid Start Byte: send sysex buffered until now, even if invalid
258  [*commandsArray addObject:[self commandByCoalescingSysexData]];
259  // Parse current packet normally
260  return NO;
261  }
262 
263  for (UInt16 idx = 0; idx < packet->length; idx++) {
264  Byte byte = data[idx];
265 
266  // Append byte
267  [self.sysexData appendBytes:&byte length:1];
268 
269  // Check for Sysex End
270  if (byte == kMIKMIDISysexEndDelimiter) {
271  [*commandsArray addObject:[self commandByCoalescingSysexData]];
272  break;
273  }
274  }
275 
276  return YES;
277 }

314 void MIKMIDIPortReadCallback(const MIDIPacketList *pktList, void *readProcRefCon, void *srcConnRefCon)
315 {
316  @autoreleasepool {
317  MIKMIDIInputPort *self = (__bridge MIKMIDIInputPort *)readProcRefCon;
318  MIKMIDISourceEndpoint *source = (__bridge MIKMIDISourceEndpoint *)srcConnRefCon;

322  }];

326 - (void)interpretPacketList:(const MIDIPacketList *)pktList handleResultingCommands:(void (^_Nonnull)(NSArray <MIKMIDICommand*> *receivedCommands))completionBlock
327 {
328  NSMutableArray *receivedCommands = [NSMutableArray array];
329 
330  // Get the first packet
331  MIDIPacket *packet = (MIDIPacket *)pktList->packet;
332 
333  for (UInt32 i = 0; i < pktList->numPackets; i++)
334  {
335  // Ignore empty packets
336  if (packet->length == 0) {
337  packet = MIDIPacketNext(packet);
338  continue;
339  }
340 
341  // Try Sysex Coalescing, otherwise parse MIDI commands
342  if (![self coalesceSysexFromMIDIPacket:packet toCommandInArray:&receivedCommands]) {
343  [receivedCommands addObjectsFromArray:[MIKMIDICommand commandsWithMIDIPacket:packet]];
344  }
345 
346  packet = MIDIPacketNext(packet);
347  }
348 
349  // Safeguard against sysex time-out
350  if (self.isCoalescingSysex) {
351  // Create or extend time-out timer
352  if (!self.sysexTimeOutTimer) {
353  // Weakify Self
354  __weak typeof(self) weakSelf = self;
355 
356  self.sysexTimeOutTimer = [NSTimer timerWithTimeInterval:self.sysexTimeOut target:[NSBlockOperation blockOperationWithBlock:^{
357  // Strongify Self
358  __strong typeof(self) self = weakSelf;
359 
360  // Force-End Sysex, if necessary
361  if (self.isCoalescingSysex) {
362  completionBlock(@[[self commandByCoalescingSysexData]]);
363  }
364  }] selector:@selector(main) userInfo:nil repeats:NO];
365 
366  // Run Timer
367  NSRunLoop *currentRunLoop = [NSRunLoop currentRunLoop];
368  NSRunLoopMode mode = currentRunLoop.currentMode ?: NSDefaultRunLoopMode;
369 
370  [currentRunLoop addTimer:self.sysexTimeOutTimer forMode:mode];
371  } else {
372  self.sysexTimeOutTimer.fireDate = [NSDate dateWithTimeIntervalSinceNow:self.sysexTimeOut];
373  }
374  return;
375  }
376 
377  // Clear Sysex Timer
378  [self.sysexTimeOutTimer invalidate];
379  self.sysexTimeOutTimer = nil;
380 
381  // Handle Commands
382  if (receivedCommands.count == 0) {
383  return;
384  }
385 
386  if (self.coalesces14BitControlChangeCommands) {
387  dispatch_sync(self.bufferedCommandQueue, ^{
388  if ([self.bufferedMSBCommands count]) {
389  [receivedCommands insertObject:self.bufferedMSBCommands.firstObject atIndex:0];
390  [self.bufferedMSBCommands removeObjectAtIndex:0];
391  }
392  });
393  receivedCommands = [[self commandsByCoalescingCommands:receivedCommands] mutableCopy];
394  MIKMIDICommand *finalCommand = [receivedCommands lastObject];
395  if ([self commandIsPossibleMSBOf14BitCommand:finalCommand]) {
396  // Hold back and wait for a possible LSB command to come in.
397  dispatch_sync(self.bufferedCommandQueue, ^{ [self.bufferedMSBCommands addObject:finalCommand]; });
398  [receivedCommands removeLastObject];
399 
400  // Wait 4ms, then send the buffered command if it hasn't been coalesced (and therefore set to nil)
401  dispatch_time_t popTime = dispatch_time(DISPATCH_TIME_NOW, (int64_t)(4 * NSEC_PER_MSEC));
402  dispatch_after(popTime, self.bufferedCommandQueue, ^(void){
403  if (![self.bufferedMSBCommands containsObject:finalCommand]) return;
404  [self.bufferedMSBCommands removeObject:finalCommand];
405  completionBlock(@[finalCommand]);
406  });
407  }
408  }
409 
410  if ([receivedCommands count] == 0) {
411  return;
412  }
413 
414  completionBlock(receivedCommands);
415 }

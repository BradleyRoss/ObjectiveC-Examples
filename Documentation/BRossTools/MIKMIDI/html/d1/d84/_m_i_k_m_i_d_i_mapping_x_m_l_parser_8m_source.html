69 - (void)parser:(NSXMLParser *)parser didStartElement:(NSString *)elementName namespaceURI:(NSString *)namespaceURI qualifiedName:(NSString *)qName attributes:(NSDictionary *)attributeDict
70 {
71  if ([elementName isEqualToString:@"Mapping"]) {
72  self.currentMapping = [[MIKMIDIMapping alloc] init];
73  self.currentMapping.controllerName = attributeDict[@"ControllerName"];
74  self.currentMapping.name = attributeDict[@"MappingName"];
75  NSMutableDictionary *attributeDictScratch = [attributeDict mutableCopy];
76  [attributeDictScratch removeObjectForKey:@"ControllerName"];
77  [attributeDictScratch removeObjectForKey:@"MappingName"];
78  self.currentMapping.additionalAttributes = attributeDictScratch;
79  return;
80  }
81 
82  if ([elementName isEqualToString:@"MappingItem"]) {
83  self.currentItemInfo = [NSMutableDictionary dictionary];
84  self.currentItemInfo[@"additionalAttributes"] = [NSMutableDictionary dictionary];
85  for (NSString *key in attributeDict) {
86  id attributeValue = attributeDict[key];
87  if ([key isEqualToString:@"InteractionType"]) {
88  self.currentItemInfo[@"interactionType"] = @(MIKMIDIMappingInteractionTypeForAttributeString(attributeValue));
89  continue;
90  } else if ([key isEqualToString:@"Flipped"]) {
91  self.currentItemInfo[@"flipped"] = @([attributeValue boolValue]);
92  continue;
93  }
94 
95  self.currentItemInfo[@"additionalAttributes"][key] = attributeValue;
96  }
97  return;
98  }
99 
100  if (self.currentItemInfo) {
101  // In the middle parsing a mapping item
102  self.currentElementName = elementName;
103  self.currentElementValueBuffer = [NSMutableString string];
104 
105  return;
106  }
107 }

114 - (void)parser:(NSXMLParser *)parser didEndElement:(NSString *)elementName namespaceURI:(NSString *)namespaceURI qualifiedName:(NSString *)qName
115 {
116  if ([elementName isEqualToString:@"Mapping"]) {
117  if (!self.currentMapping) return;
118  [self.internalMappings addObject:self.currentMapping];
119  self.currentMapping = nil;
120 
121  return;
122  }
123 
124  if ([elementName isEqualToString:@"MappingItem"]) {
125  NSString *responderID = self.currentItemInfo[@"responderIdentifier"];
126  [self.currentItemInfo removeObjectForKey:@"responderIdentifier"];
127  NSString *commandID = self.currentItemInfo[@"commandIdentifier"];
128  [self.currentItemInfo removeObjectForKey:@"commandIdentifier"];
129  MIKMIDIMappingItem *item = [[MIKMIDIMappingItem alloc] initWithMIDIResponderIdentifier:responderID andCommandIdentifier:commandID];
130 
131  item.additionalAttributes = self.currentItemInfo[@"additionalAttributes"];
132  item.channel = [self.currentItemInfo[@"channel"] integerValue];
133  item.commandType = [self.currentItemInfo[@"commandType"] integerValue];
134  item.controlNumber = [self.currentItemInfo[@"controlNumber"] unsignedIntegerValue];
135  item.flipped = [self.currentItemInfo[@"flipped"] boolValue];
136  item.interactionType = [self.currentItemInfo[@"interactionType"] integerValue];
137 
138  [self.currentMapping addMappingItemsObject:item];
139 
140  self.currentItemInfo = nil;
141 
142  return;
143  }
144 
145  if ([elementName isEqualToString:self.currentElementName]) {
146  // Current element is stored as a string which may need to be converted to an NSNumber.
147  // Using NSScanner here, as detailed in https://stackoverflow.com/a/572312/8653957
148  NSString* currentElement = [self.currentElementValueBuffer copy];
149  NSScanner* scanner = [NSScanner scannerWithString:currentElement];
150  self.currentItemInfo[[elementName mik_uncapitalizedString]] = ([scanner scanInt:nil] && [scanner isAtEnd]) ? @(currentElement.integerValue) : currentElement;
151 
152  self.currentElementName = nil;
153  self.currentElementValueBuffer = nil;
154 
155  return;
156  }

182 - (NSString *)mik_uncapitalizedString
183 {
184  // This is a bit quick and dirty, but it does the job
185  if (![self length]) return self;
186 
187  NSMutableString *scratch = [self mutableCopy];
188  NSString *firstCharacter = [self substringToIndex:1];
189  [scratch replaceCharactersInRange:NSMakeRange(0, 1) withString:[firstCharacter lowercaseString]];

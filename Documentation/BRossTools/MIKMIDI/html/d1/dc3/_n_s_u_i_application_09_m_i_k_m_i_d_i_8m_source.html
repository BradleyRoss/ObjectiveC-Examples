39 static BOOL MIKObjectRespondsToMIDICommand(id object, MIKMIDICommand *command)

112 
113  id<MIKMIDIResponder> result = nil;

137 - (NSSet *)recursiveSubrespondersOfMIDIResponder:(id<MIKMIDIResponder>)responder
138 {
139  NSMutableSet *result = [NSMutableSet setWithObject:responder];
140  if (![responder respondsToSelector:@selector(subresponders)]) return result;
141 
142  NSArray *subresponders = [responder subresponders];
143  [result addObjectsFromArray:subresponders];
144  for (id<MIKMIDIResponder>subresponder in subresponders) {

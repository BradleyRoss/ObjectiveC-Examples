185 - (void)addClickTrackWhenNeededFromTimeStamp:(MusicTimeStamp)fromTimeStamp
186 {
187  if (self.isClickPlayer) return;
188  if (!self.isClickTrackEnabled) return;
189 
190  self.clickPlayer = [[MIKMIDIPlayer alloc] init];
191  self.clickPlayer->_isClickPlayer = YES;
192  MIKMIDISequence *clickSequence = [MIKMIDISequence sequence];
193  [clickSequence.tempoTrack addEvents:self.sequence.tempoEvents];
194  [clickSequence.tempoTrack addEvents:self.sequence.timeSignatureEvents];
195  self.clickPlayer.sequence = clickSequence;
196  MIKMIDITrack *clickTrack = [clickSequence addTrackWithError:NULL];
197 
198  OSStatus err = MusicTrackSetDestMIDIEndpoint(clickTrack.musicTrack, (MIDIEndpointRef)self.metronomeEndpoint.objectRef);
199  if (err) {
200  NSLog(@"MusicTrackGetProperty() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
201  return;
202  }
203  MusicTimeStamp toTimeStamp = self.stopPlaybackAtEndOfSequence ? self.sequence.length : self.maxClickTrackTimeStamp;
204 
205  NSMutableSet *clickEvents = [NSMutableSet set];
206  MIDINoteMessage tickMessage = self.metronome.tickMessage;
207  MIDINoteMessage tockMessage = self.metronome.tockMessage;
208  MusicTimeStamp increment = 1;
209  for (MusicTimeStamp clickTimeStamp = floor(fromTimeStamp); clickTimeStamp <= toTimeStamp; clickTimeStamp += increment) {
210  MIKMIDITimeSignature timeSignature = [clickSequence timeSignatureAtTimeStamp:clickTimeStamp];
211  if (!timeSignature.numerator || !timeSignature.denominator) continue;
212 
213  NSInteger adjustedTimeStamp = (NSInteger)(clickTimeStamp * timeSignature.denominator / 4.0);
214  BOOL isTick = !((adjustedTimeStamp + timeSignature.numerator) % (timeSignature.numerator));
215  increment = 4.0 / timeSignature.denominator;
216 
217  MIDINoteMessage clickMessage = isTick ? tickMessage : tockMessage;
218  [clickEvents addObject:[MIKMIDINoteEvent noteEventWithTimeStamp:clickTimeStamp message:clickMessage]];
219  }
220 
221  [clickTrack addEvents:[clickEvents allObjects]];
222 }

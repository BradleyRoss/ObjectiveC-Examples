100 - (UInt32)manufacturerID
101 {
102  if ([self.internalData length] < 2) return 0;
103 
104  NSUInteger manufacturerIDLength = _has3ByteManufacturerID ? 3 : 1;
105  NSData *idData = [self.internalData subdataWithRange:NSMakeRange(1, manufacturerIDLength)];
106  UInt8 *bytes = (UInt8 *)[idData bytes];
107  if (manufacturerIDLength == 1) { return bytes[0]; }
108  return (UInt32)(bytes[0] << 16 | bytes[1] << 8 | bytes[2]);
109 }

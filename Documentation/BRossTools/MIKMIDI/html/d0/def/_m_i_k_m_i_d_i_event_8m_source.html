145 + (MIKMIDIEventType)mikEventTypeForMusicEventType:(MusicEventType)musicEventType andData:(NSData *)data
146 {
147  static NSDictionary *channelEventTypeToMIDITypeMap = nil;
148  static NSDictionary *musicEventToMIDITypeMap = nil;
149  static dispatch_once_t onceToken;
150  dispatch_once(&onceToken, ^{
151  channelEventTypeToMIDITypeMap = @{@(MIKMIDIChannelEventTypePolyphonicKeyPressure) : @(MIKMIDIEventTypeMIDIPolyphonicKeyPressureMessage),
152  @(MIKMIDIChannelEventTypeControlChange) : @(MIKMIDIEventTypeMIDIControlChangeMessage),
153  @(MIKMIDIChannelEventTypeProgramChange) : @(MIKMIDIEventTypeMIDIProgramChangeMessage),
154  @(MIKMIDIChannelEventTypeChannelPressure) : @(MIKMIDIEventTypeMIDIChannelPressureMessage),
155  @(MIKMIDIChannelEventTypePitchBendChange) : @(MIKMIDIEventTypeMIDIPitchBendChangeMessage)};
156 
157  musicEventToMIDITypeMap = @{@(kMusicEventType_NULL) : @(MIKMIDIEventTypeNULL),
158  @(kMusicEventType_ExtendedNote) : @(MIKMIDIEventTypeExtendedNote),
159  @(kMusicEventType_ExtendedTempo) : @(MIKMIDIEventTypeExtendedTempo),
160  @(kMusicEventType_User) : @(MIKMIDIEventTypeUser),
161  @(kMusicEventType_Meta) : @(MIKMIDIEventTypeMeta),
162  @(kMusicEventType_MIDINoteMessage) : @(MIKMIDIEventTypeMIDINoteMessage),
163  @(kMusicEventType_MIDIChannelMessage) : @(MIKMIDIEventTypeMIDIChannelMessage),
164  @(kMusicEventType_MIDIRawData) : @(MIKMIDIEventTypeMIDIRawData),
165  @(kMusicEventType_Parameter) : @(MIKMIDIEventTypeParameter),
166  @(kMusicEventType_AUPreset) : @(MIKMIDIEventTypeAUPreset),};
167 
168  });
169 
170  if (musicEventType == kMusicEventType_Meta) {
171  MIKMIDIMetaEventType metaEventType = *(MIKMIDIMetaEventType *)[data bytes];
172  return [MIKMIDIMetaEvent eventTypeForMetaSubtype:metaEventType];
173  } else if (musicEventType == kMusicEventType_MIDIChannelMessage) {
174  UInt8 channelEventType = *(UInt8 *)[data bytes] & 0xF0;
175  return [channelEventTypeToMIDITypeMap[@(channelEventType)] unsignedIntegerValue];
176  } else {
177  return [musicEventToMIDITypeMap[@(musicEventType)] unsignedIntegerValue];
178  }
179 }

144  [attributeElement setStringValue:stringValue];
145  [attributes addObject:attributeElement];
146  }

149  NSSortDescriptor *sortByCommandID = [NSSortDescriptor sortDescriptorWithKey:@"commandIdentifier" ascending:YES];
150  NSArray *sortedMappingItems = [self.mappingItems sortedArrayUsingDescriptors:@[sortByResponderID, sortByCommandID]];
151  NSArray *mappingItemXMLElements = [sortedMappingItems valueForKey:@"XMLRepresentation"];

166 - (NSString *)XMLStringRepresentation;
167 {
168 #if !TARGET_OS_IPHONE
169  return [[self privateXMLRepresentation] XMLStringWithOptions:NSXMLNodePrettyPrint];
170 #else
171 
172  int err = 0;
173  xmlTextWriterPtr writer = NULL;
174  xmlBufferPtr buffer = xmlBufferCreate();
175  if (!buffer) {
176  NSLog(@"Unable to create XML buffer.");
177  goto CLEANUP_AND_EXIT;
178  }
179 
180  writer = xmlNewTextWriterMemory(buffer, 0);
181  if (!writer) {
182  xmlBufferFree(buffer);
183  NSLog(@"Unable to create XML writer.");
184  goto CLEANUP_AND_EXIT;
185  }
186 
187  // Start the document
188  err = xmlTextWriterStartDocument(writer, NULL, "UTF-8", NULL);
189  if (err < 0) {
190  NSLog(@"Unable to start XML document: %i", err);
191  goto CLEANUP_AND_EXIT;
192  }
193 
194  err = xmlTextWriterStartElement(writer, BAD_CAST "Mapping"); // <Mapping>
195  if (err < 0) {
196  NSLog(@"Unable to start XML Mapping element: %i", err);
197  goto CLEANUP_AND_EXIT;
198  }
199 
200  xmlTextWriterSetIndent(writer, 1);
201 
202  err = xmlTextWriterWriteAttribute(writer, BAD_CAST "ControllerName", BAD_CAST [self.controllerName UTF8String]);
203  if (err < 0) {
204  NSLog(@"Unable to write ControllerName attribute for Mapping element: %i", err);
205  goto CLEANUP_AND_EXIT;
206  }
207 
208  err = xmlTextWriterWriteAttribute(writer, BAD_CAST "MappingName", BAD_CAST [self.name UTF8String]);
209  if (err < 0) {
210  NSLog(@"Unable to write MappingName attribute for Mapping element: %i", err);
211  goto CLEANUP_AND_EXIT;
212  }
213 
214  for (NSString *key in self.additionalAttributes) {
215  NSString *stringValue = self.additionalAttributes[key];
216  if (![stringValue isKindOfClass:[NSString class]]) {
217  NSLog(@"Ignoring additional attribute %@ : %@ because it is not a string.", key, stringValue);
218  continue;
219  }
220 
221  err = xmlTextWriterWriteAttribute(writer, BAD_CAST [key UTF8String], BAD_CAST [stringValue UTF8String]);
222  if (err < 0) {
223  NSLog(@"Unable to write MappingName attribute for Mapping element: %i", err);
224  goto CLEANUP_AND_EXIT;
225  }
226  }
227 
228  err = xmlTextWriterStartElement(writer, BAD_CAST "MappingItems"); // <MappingItems>
229  if (err < 0) {
230  NSLog(@"Unable to start XML Mapping Items element: %i", err);
231  goto CLEANUP_AND_EXIT;
232  }
233 
234  {
235  // Write mapping items
236  NSSortDescriptor *sortByResponderID = [NSSortDescriptor sortDescriptorWithKey:@"MIDIResponderIdentifier" ascending:YES];
237  NSSortDescriptor *sortByCommandID = [NSSortDescriptor sortDescriptorWithKey:@"commandIdentifier" ascending:YES];
238  NSArray *sortedMappingItems = [self.mappingItems sortedArrayUsingDescriptors:@[sortByResponderID, sortByCommandID]];
239 
240  for (MIKMIDIMappingItem *item in sortedMappingItems) {
241  NSString *xmlString = [item XMLStringRepresentation];
242  err = xmlTextWriterWriteRaw(writer, BAD_CAST [xmlString UTF8String]);
243  if (err < 0) {
244  NSLog(@"Unable to write XML for mapping item %@: %i", item, err);
245  goto CLEANUP_AND_EXIT;
246  }
247  }
248 
249  err = xmlTextWriterEndElement(writer); // </MappingItems>
250  if (err < 0) {
251  NSLog(@"Unable to end XML Mapping Items element: %i", err);
252  goto CLEANUP_AND_EXIT;
253  }
254 
255  err = xmlTextWriterEndElement(writer); // </Mapping>
256  if (err < 0) {
257  NSLog(@"Unable to end XML Mapping element: %i", err);
258  goto CLEANUP_AND_EXIT;
259  }
260 
261  err = xmlTextWriterEndDocument(writer);
262  if (err < 0) {
263  NSLog(@"Unable to end XML Mapping document: %i", err);
264  goto CLEANUP_AND_EXIT;
265  }
266  }
267 
268 CLEANUP_AND_EXIT:
269  if (writer) xmlFreeTextWriter(writer);
270  NSString *result = nil;
271  if (buffer && err >= 0) {
272  result = [[NSString alloc] initWithCString:(const char *)buffer->content encoding:NSUTF8StringEncoding];
273  xmlBufferFree(buffer);
274  }
275 
276  return result;

368 - (BOOL)loadPropertiesFromXMLDocument:(NSXMLDocument *)xmlDocument
369 {
370  NSError *error = nil;
371 
372  NSArray *mappings = [xmlDocument nodesForXPath:@"./Mapping" error:&error];
373  if (![mappings count]) {
374  NSLog(@"Unable to get mapping from MIDI Mapping XML: %@", error);
375  return NO;
376  }
377  NSXMLElement *mapping = [mappings lastObject];
378 
379  NSArray *nameAttributes = [mapping nodesForXPath:@"./@MappingName" error:&error];
380  if (!nameAttributes) NSLog(@"Unable to get name attributes from MIDI Mapping XML: %@", error);
381  self.name = [[nameAttributes lastObject] stringValue];
382 
383  NSArray *controllerNameAttributes = [mapping nodesForXPath:@"./@ControllerName" error:&error];
384  if (!controllerNameAttributes) NSLog(@"Unable to get controller name attributes from MIDI Mapping XML: %@", error);
385  self.controllerName = [[controllerNameAttributes lastObject] stringValue];
386 
387  NSArray *mappingItemElements = [mapping nodesForXPath:@"./MappingItems/MappingItem" error:&error];
388  if (!mappingItemElements) {
389  NSLog(@"Unable to get mapping items from MIDI mapping XML: %@", error);
390  return NO;
391  }
392 
393  for (NSXMLElement *element in mappingItemElements) {
394  MIKMIDIMappingItem *item = [[MIKMIDIMappingItem alloc] initWithXMLElement:element];
395  if (!item) continue;
396  [self addMappingItemsObject:item];
397  }
398 
399  NSMutableDictionary *attributes = [NSMutableDictionary dictionary];
400  for (NSXMLNode *attribute in [mapping attributes]) {
401  if (![[attribute stringValue] length]) continue;
402  if ([[attribute name] isEqualToString:@"MappingName"]) continue;
403  if ([[attribute name] isEqualToString:@"ControllerName"]) continue;
404  [attributes setObject:[attribute stringValue] forKey:[attribute name]];
405  }
406  self.additionalAttributes = attributes;
407 
408  return YES;
409 }

111 - (void)retrieveEndpoints
112 {
113  NSMutableArray *sources = [NSMutableArray array];
114  ItemCount numSources = MIDIEntityGetNumberOfSources(self.objectRef);
115  for (ItemCount i=0; i<numSources; i++) {
116  MIDIEndpointRef sourceRef = MIDIEntityGetSource(self.objectRef, i);
117  MIKMIDISourceEndpoint *source = [MIKMIDISourceEndpoint MIDIObjectWithObjectRef:sourceRef];
118  if (!source) continue;
119  source.entity = self;
120  [sources addObject:source];
121  }
122  self.internalSources = sources;
123 
124  NSMutableArray *destinations = [NSMutableArray array];
125  ItemCount numDestinations = MIDIEntityGetNumberOfDestinations(self.objectRef);
126  for (ItemCount i=0; i<numDestinations; i++) {
127  MIDIEndpointRef destinationRef = MIDIEntityGetDestination(self.objectRef, i);
128  MIKMIDISourceEndpoint *destination = [MIKMIDISourceEndpoint MIDIObjectWithObjectRef:destinationRef];
129  if (!destination) continue;
130  destination.entity = self;
131  [destinations addObject:destination];
132  }
133  self.internalDestinations = destinations;
134 }

36 - (BOOL)sendCommands:(NSArray *)commands toDestination:(MIKMIDIDestinationEndpoint *)destination error:(NSError **)error;
37 {
38  commands = [self commandsByTransformingForTransmissionCommands:commands];
39  if (![commands count] || !destination) return NO;
40 
41  error = error ? error : &(NSError *__autoreleasing){ nil };
42 
43  MIDIPacketList *packetList;
44  if (!MIKCreateMIDIPacketListFromCommands(&packetList, commands)) return NO;
45 
46  OSStatus err = MIDISend(self.portRef, destination.objectRef, packetList);
47  free(packetList);
48  if (err != noErr) {
49  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
50  return NO;
51  }

59 - (NSArray *)commandsByTransformingForTransmissionCommands:(NSArray *)commands
60 {

63  if ([command respondsToSelector:@selector(commandsForTransmission)]) {
64  [transformedCommands addObjectsFromArray:[command commandsForTransmission]];

66  [transformedCommands addObject:command];

54 + (instancetype)MIDIObjectWithObjectRef:(MIDIObjectRef)objectRef;
55 {
56  if (objectRef == 0) return nil;
57  Class resultSubclass = nil;
58  for (Class subclass in registeredMIKMIDIObjectSubclasses) {
59  if ([subclass canInitWithObjectRef:objectRef]) {
60  resultSubclass = subclass;
61  break;
62  }
63  }
64  if (!resultSubclass) return nil;
65 
66  return [[resultSubclass alloc] initWithObjectRef:objectRef];
67 }

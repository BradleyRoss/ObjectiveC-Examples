138 - (void)saveConfiguration
139 {
140  NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
141 
142  NSMutableDictionary *configuration = [NSMutableDictionary dictionaryWithDictionary:[self savedConfiguration]];
143 
144  // Save connected device names
145  NSMutableArray *connectedDeviceNames = [configuration[MIKMIDIConnectionManagerConnectedDevicesKey] mutableCopy];
146  if (!connectedDeviceNames) {
147  connectedDeviceNames = [NSMutableArray array];
148  configuration[MIKMIDIConnectionManagerConnectedDevicesKey] = connectedDeviceNames;
149  }
150 
151  // And explicitly unconnected device names
152  NSMutableArray *unconnectedDeviceNames = [configuration[MIKMIDIConnectionManagerUnconnectedDevicesKey] mutableCopy];
153  if (!unconnectedDeviceNames) {
154  unconnectedDeviceNames = [NSMutableArray array];
155  configuration[MIKMIDIConnectionManagerUnconnectedDevicesKey] = unconnectedDeviceNames;
156  }
157 
158  // For devices that were connected in saved configuration but are now unavailable, leave them
159  // connected in the configuration so they'll reconnect automatically.
160  for (MIKMIDIDevice *device in self.availableDevices) {
161  NSString *name = device.name;
162  if (![name length]) continue;
163  if ([self isConnectedToDevice:device]) {
164  if (![connectedDeviceNames containsObject:name]) { [connectedDeviceNames addObject:name]; }
165  [unconnectedDeviceNames removeObject:name];
166  } else {
167  [connectedDeviceNames removeObject:name];
168  if (![unconnectedDeviceNames containsObject:name]) { [unconnectedDeviceNames addObject:name]; }
169  }
170  }
171 
172  configuration[MIKMIDIConnectionManagerConnectedDevicesKey] = connectedDeviceNames;
173  configuration[MIKMIDIConnectionManagerUnconnectedDevicesKey] = unconnectedDeviceNames;
174 
175  [userDefaults setObject:configuration forKey:[self userDefaultsConfigurationKey]];
176 }

195 - (void)updateAvailableDevices
196 {
197  NSArray *regularDevices = self.deviceManager.availableDevices;
198  NSMutableArray *result = [NSMutableArray arrayWithArray:regularDevices];
199 
200  if (self.includesVirtualDevices) {
201  NSMutableSet *endpointsInDevices = [NSMutableSet set];
202  for (MIKMIDIDevice *device in regularDevices) {
203  NSSet *sources = [NSSet setWithArray:[device.entities valueForKeyPath:@"@distinctUnionOfArrays.sources"]];
204  NSSet *destinations = [NSSet setWithArray:[device.entities valueForKeyPath:@"@distinctUnionOfArrays.destinations"]];
205  [endpointsInDevices unionSet:sources];
206  [endpointsInDevices unionSet:destinations];
207  }
208 
209  NSMutableSet *devicelessSources = [NSMutableSet setWithArray:self.deviceManager.virtualSources];
210  NSMutableSet *devicelessDestinations = [NSMutableSet setWithArray:self.deviceManager.virtualDestinations];
211  [devicelessSources minusSet:endpointsInDevices];
212  [devicelessDestinations minusSet:endpointsInDevices];
213 
214  // Now we need to try to associate each source with its corresponding destination on the same device
215  NSMapTable *destinationToSourceMap = [NSMapTable mapTableWithKeyOptions:NSMapTableStrongMemory valueOptions:NSMapTableStrongMemory];
216  NSMapTable *deviceNamesBySource = [NSMapTable mapTableWithKeyOptions:NSMapTableStrongMemory valueOptions:NSMapTableStrongMemory];
217 
218  for (MIKMIDIEndpoint *source in devicelessSources) {
219  NSString *sourceName = [self deviceNameFromVirtualEndpoint:source];
220  for (MIKMIDIEndpoint *destination in devicelessDestinations) {
221  NSString *destinationName = [self deviceNameFromVirtualEndpoint:destination];
222  if ([sourceName isEqualToString:destinationName]) { // Source and destination match
223  [destinationToSourceMap setObject:destination forKey:source];
224  [deviceNamesBySource setObject:sourceName forKey:source];
225  break;
226  }
227  }
228  }
229 
230  for (MIKMIDIEndpoint *source in destinationToSourceMap) {
231  MIKMIDIEndpoint *destination = [destinationToSourceMap objectForKey:source];
232  [devicelessSources removeObject:source];
233  [devicelessDestinations removeObject:destination];
234 
235  MIKMIDIDevice *device = [MIKMIDIDevice deviceWithVirtualEndpoints:@[source, destination]];
236  device.name = [deviceNamesBySource objectForKey:source];
237  if (device) [result addObject:device];
238  }
239  for (MIKMIDIEndpoint *endpoint in devicelessSources) {
240  MIKMIDIDevice *device = [MIKMIDIDevice deviceWithVirtualEndpoints:@[endpoint]];
241  if (device) [result addObject:device];
242  }
243  }
244 
245  self.availableDevices = [result copy];
246 }

256 - (BOOL)internalConnectToDevice:(MIKMIDIDevice *)device error:(NSError **)error
257 {
258  if ([self isConnectedToDevice:device]) return YES;
259  error = error ?: &(NSError *__autoreleasing){ nil };
260 
261  __weak typeof(self) weakSelf = self;
262  id token = [self.deviceManager connectDevice:device error:error eventHandler:^(MIKMIDISourceEndpoint *endpoint, NSArray *commands) {
263  __strong typeof(self) strongSelf = weakSelf;
264  if (!strongSelf) { return; } // shouldn't actually happen
265  [strongSelf recordPendingNoteOnCommands:commands fromDevice:device];
266  [strongSelf removePendingNoteOnCommandsTerminatedByNoteOffCommands:commands fromDevice:device];
267 
268  MIKMIDIEventHandlerBlock eventHandler = [strongSelf eventHandler];
269  if (eventHandler) { eventHandler(endpoint, commands); }
270  }];
271  if (!token) return NO;
272 
273  [self.connectionTokensByDevice setObject:token forKey:device];
274  [self willChangeValueForKey:@"connectedDevices"
275  withSetMutation:NSKeyValueUnionSetMutation
276  usingObjects:[NSSet setWithObject:device]];
277  [self.internalConnectedDevices addObject:device];
278  [self didChangeValueForKey:@"connectedDevices"
279  withSetMutation:NSKeyValueUnionSetMutation
280  usingObjects:[NSSet setWithObject:device]];
281 
282  __strong typeof(_delegate) delegate = self.delegate;
283  if ([delegate respondsToSelector:@selector(connectionManager:deviceWasConnected:)]) {
284  [delegate connectionManager:self deviceWasConnected:device];
285  }
286 
287  return YES;
288 }

290 - (void)internalDisconnectFromDevice:(MIKMIDIDevice *)device
291 {
292  if (![self isConnectedToDevice:device]) return;
293 
294  id token = [self.connectionTokensByDevice objectForKey:device];
295  if (!token) return;
296 
297  [self.deviceManager disconnectConnectionForToken:token];
298 
299  [self.connectionTokensByDevice removeObjectForKey:device];
300  [self willChangeValueForKey:@"connectedDevices"
301  withSetMutation:NSKeyValueMinusSetMutation
302  usingObjects:[NSSet setWithObject:device]];
303  [self.internalConnectedDevices removeObject:device];
304  [self didChangeValueForKey:@"connectedDevices"
305  withSetMutation:NSKeyValueMinusSetMutation
306  usingObjects:[NSSet setWithObject:device]];
307 
308  __strong typeof(_delegate) delegate = self.delegate;
309  if ([delegate respondsToSelector:@selector(connectionManager:deviceWasDisconnected:withUnterminatedNoteOnCommands:)]) {
310  NSArray *pendingNoteOns = [self pendingNoteOnCommandsForDevice:device];
311  [delegate connectionManager:self deviceWasDisconnected:device withUnterminatedNoteOnCommands:pendingNoteOns];
312  }
313 
314  if (self.automaticallySavesConfiguration) [self saveConfiguration];

324 - (void)connectToNewlyAddedDeviceIfAppropriate:(MIKMIDIDevice *)device
325 {
326  if (!device) return;
327 
328  MIKMIDIAutoConnectBehavior behavior = MIKMIDIAutoConnectBehaviorConnectIfPreviouslyConnectedOrNew;
329 
330  __strong typeof(_delegate) delegate = self.delegate;
331  if ([delegate respondsToSelector:@selector(connectionManager:shouldConnectToNewlyAddedDevice:)]) {
332  behavior = [delegate connectionManager:self shouldConnectToNewlyAddedDevice:device];
333  }
334 
335  BOOL shouldConnect = NO;
336  switch (behavior) {
337  case MIKMIDIAutoConnectBehaviorDoNotConnect:
338  shouldConnect = NO;
339  break;
340  case MIKMIDIAutoConnectBehaviorConnect:
341  shouldConnect = YES;
342  break;
343  case MIKMIDIAutoConnectBehaviorConnectOnlyIfPreviouslyConnected:
344  shouldConnect = [self deviceIsConnectedInSavedConfiguration:device];
345  break;
346  case MIKMIDIAutoConnectBehaviorConnectIfPreviouslyConnectedOrNew:
347  shouldConnect = ![self deviceIsUnconnectedInSavedConfiguration:device];
348  break;
349  }
350 
351  if (shouldConnect) {
352  NSError *error = nil;
353  if (![self internalConnectToDevice:device error:&error]) {
354  NSLog(@"Unable to connect to MIDI device %@: %@", device, error);
355  return;
356  }
357  }

443 - (void)removePendingNoteOnCommandsTerminatedByNoteOffCommands:(MIKArrayOf(MIKMIDICommand *) *)commands fromDevice:(MIKMIDIDevice *)device
444 {
445  commands = [commands filteredArrayUsingPredicate:[NSPredicate predicateWithBlock:^BOOL(id obj, NSDictionary *b) {
446  if ([obj isKindOfClass:[MIKMIDINoteOnCommand class]] &&
447  [(MIKMIDINoteOnCommand *)obj velocity] == 0) {
448  return YES;
449  }
450  return [obj isKindOfClass:[MIKMIDINoteOffCommand class]];
451  }]];
452  if (![commands count]) return;
453 
454  NSMutableArray *pendingNoteOns = [self pendingNoteOnCommandsForDevice:device];
455  if (![pendingNoteOns count]) return;
456 
457  for (MIKMIDINoteOffCommand *noteOff in commands) {
458  for (MIKMIDINoteOnCommand *noteOn in [pendingNoteOns copy]) {
459  if (MIKMIDINoteOffCommandCorrespondsWithNoteOnCommand(noteOff, noteOn)) {
460  [pendingNoteOns removeObject:noteOn];

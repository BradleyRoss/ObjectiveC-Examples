54 {
55  if (!msbCommand || !lsbCommand) return nil;
56 
57  if (![msbCommand isKindOfClass:[MIKMIDIControlChangeCommand class]] ||
58  ![lsbCommand isKindOfClass:[MIKMIDIControlChangeCommand class]]) return nil;
59 
60  if (msbCommand.controllerNumber > 31) return nil;

169 - (NSUInteger)fourteenBitValue
170 {
171  NSUInteger MSB = ([super value] << 7) & 0x3F80;
172  NSUInteger LSB = 0;
173  if ([self.data length] > 3) {

182 {

184 
185  NSUInteger MSB = (value >> 7) & 0x7F;

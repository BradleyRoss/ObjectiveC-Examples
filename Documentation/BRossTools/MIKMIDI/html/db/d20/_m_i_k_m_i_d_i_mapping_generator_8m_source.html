165 - (void)handleMIDICommand:(MIKMIDIChannelVoiceCommand *)command
166 {
167  if (self.isMappingSuspended) return; // Ignore input while mapping is suspended
168 
169  if ([self.receivedMessages count]) {
170  // If we get a message from a different controller number, channel,
171  // or command type (not counting note on vs note off), restart the mapping
172 
173  BOOL allowTouchSenseMessages = NO;
174 
175  // Ignore different message types if we're trying to map a knob,
176  // since they sometimes send note on/off commands for touch sensing.
177  if (self.responderTypeOfControlBeingLearned & MIKMIDIResponderTypeKnob &&
178  [self.receivedMessages count] > [self defaultMinimumNumberOfMessagesRequiredForResponderType:self.responderTypeOfControlBeingLearned]) {
179  allowTouchSenseMessages = YES;
180  }
181 
182  MIKMIDIChannelVoiceCommand *firstMessage = [self.receivedMessages objectAtIndex:0];
183  BOOL firstMessageIsDifferent = ![self command:firstMessage isSameTypeChannelNumberAsCommand:command];
184  if (!allowTouchSenseMessages && firstMessageIsDifferent) {
185  [self.receivedMessages removeAllObjects];
186  }
187  }
188 
189  if (![self.controlBeingLearned respondsToMIDICommand:command]) return;
190 
191  [self.receivedMessages addObject:command];
192  self.messagesTimeoutTimer = [NSTimer scheduledTimerWithTimeInterval:self.timeoutInteveral
193  target:self
194  selector:@selector(timeoutTimerFired:)
195  userInfo:nil
196  repeats:NO];
197 }

201 - (BOOL)fillInButtonMappingItem:(MIKMIDIMappingItem **)mappingItem fromMessages:(NSArray *)messages
202 {
203  if (![messages count]) return NO;
204  if ([messages count] > 2) return NO;
205 
206  MIKMIDIChannelVoiceCommand *firstMessage = [messages objectAtIndex:0];
207 
208  MIKMIDIMappingItem *result = *mappingItem;
209  result.channel = firstMessage.channel;
210  result.controlNumber = MIKMIDIControlNumberFromCommand(firstMessage);
211 
212  // Tap type button
213  if ([messages count] == 1) {
214  if ([[NSDate date] timeIntervalSinceDate:firstMessage.timestamp] < self.timeoutInteveral) return NO; // Need to keep waiting for another message
215 
216  result.interactionType = MIKMIDIResponderTypePressButton;
217  }
218 
219  // Key type button
220  if ([messages count] == 2) {
221  MIKMIDIChannelVoiceCommand *secondMessage = [messages objectAtIndex:1];
222  BOOL firstIsZero = MIKMIDIControlValueFromChannelVoiceCommand(firstMessage) == 0 || firstMessage.commandType == MIKMIDICommandTypeNoteOff;
223  BOOL secondIsZero = MIKMIDIControlValueFromChannelVoiceCommand(secondMessage) == 0 || secondMessage.commandType == MIKMIDICommandTypeNoteOff;
224 
225  result.interactionType = (!firstIsZero && secondIsZero) ? MIKMIDIResponderTypePressReleaseButton : MIKMIDIResponderTypePressButton;
226  }
227 
228  return YES;
229 }

231 - (BOOL)fillInRelativeKnobMappingItem:(MIKMIDIMappingItem **)mappingItem fromMessages:(NSArray *)messages
232 {
233  if ([messages count] < [self defaultMinimumNumberOfMessagesRequiredForResponderType:MIKMIDIResponderTypeRelativeKnob]) return NO;
234 
235  // Disallow non-control change messages
236  for (MIKMIDIChannelVoiceCommand *message in messages) { if (message.commandType != MIKMIDICommandTypeControlChange) return NO; }
237 
238  NSMutableSet *messageValues = [NSMutableSet set];
239  for (MIKMIDIChannelVoiceCommand *message in messages) {
240  [messageValues addObject:@(MIKMIDIControlValueFromChannelVoiceCommand(message))];
241  }
242  // If there are more than 2 message values, it's more likely an absolute knob.
243  if ([messages count] == [messageValues count] || [messageValues count] > 2) return NO;
244 
245  MIKMIDIChannelVoiceCommand *firstMessage = [messages objectAtIndex:0];
246 
247  MIKMIDIMappingItem *result = *mappingItem;
248  result.interactionType = MIKMIDIResponderTypeRelativeKnob;
249  result.channel = firstMessage.channel;
250  result.controlNumber = MIKMIDIControlNumberFromCommand(firstMessage);
251  result.flipped = ([(MIKMIDIChannelVoiceCommand *)[messages lastObject] value] < 64);
252  return YES;
253 }

255 - (BOOL)fillInTurntableKnobMappingItem:(MIKMIDIMappingItem **)mappingItem fromMessages:(NSArray *)messages
256 {
257  // Filter non-control change messages
258  NSPredicate *controlChangePredicate = [NSPredicate predicateWithFormat:@"commandType == %@", @(MIKMIDICommandTypeControlChange)];
259  messages = [messages filteredArrayUsingPredicate:controlChangePredicate];
260 
261  if ([messages count] < [self defaultMinimumNumberOfMessagesRequiredForResponderType:MIKMIDIResponderTypeTurntableKnob]) return NO;
262 
263  MIKMIDIControlChangeCommand *firstMessage = [messages firstObject];
264 
265  float avgChangePerMessage = 0;
266  int maxChangePerMessage = 0;
267  int lastValue = (int)[firstMessage controllerValue];
268  for (MIKMIDIControlChangeCommand *command in messages) {
269  int change = (int)command.value - lastValue;
270  avgChangePerMessage += (float)change / (float)[messages count];
271  maxChangePerMessage = MAX(abs(change), maxChangePerMessage);
272  lastValue = (int)command.value;
273  }
274 
275  if (fabsf(avgChangePerMessage) > 0.9 && maxChangePerMessage < 63) return NO; // Probably not a turntable, more likely an absolute knob
276 
277  MIKMIDIMappingItem *result = *mappingItem;
278  result.interactionType = MIKMIDIResponderTypeTurntableKnob;
279  result.channel = firstMessage.channel;
280  result.controlNumber = MIKMIDIControlNumberFromCommand(firstMessage);
281  result.flipped = ([(MIKMIDIControlChangeCommand *)[messages lastObject] value] < 64);
282  return YES;
283 }

285 - (BOOL)fillInAbsoluteKnobSliderMappingItem:(MIKMIDIMappingItem **)mappingItem fromMessages:(NSArray *)messages
286 {
287  if ([messages count] < [self defaultMinimumNumberOfMessagesRequiredForResponderType:MIKMIDIResponderTypeAbsoluteSliderOrKnob]) return NO;
288 
289  // Disallow non-control change messages
290  for (MIKMIDIChannelVoiceCommand *message in messages) { if (message.commandType != MIKMIDICommandTypeControlChange) return NO; }
291 
292  MIKMIDIChannelVoiceCommand *firstMessage = [messages objectAtIndex:0];
293  MIKMIDIMappingItem *result = *mappingItem;
294  result.interactionType = MIKMIDIResponderTypeAbsoluteSliderOrKnob;
295  result.channel = firstMessage.channel;
296  result.controlNumber = MIKMIDIControlNumberFromCommand(firstMessage);
297 
298  // Figure out which direction it goes
299  NSInteger directionCounter = 0;
300  MIKMIDIChannelVoiceCommand *previousMessage = (MIKMIDIChannelVoiceCommand *)firstMessage;
301  for (MIKMIDIChannelVoiceCommand *message in messages) {
302  if (MIKMIDIControlValueFromChannelVoiceCommand(message) > MIKMIDIControlValueFromChannelVoiceCommand(previousMessage)) directionCounter++;
303  if (MIKMIDIControlValueFromChannelVoiceCommand(message) < MIKMIDIControlValueFromChannelVoiceCommand(previousMessage)) directionCounter--;
304  previousMessage = message;
305  }
306  result.flipped = (directionCounter < 0);
307 
308  return YES;
309 }

311 - (BOOL)fillInRelativeAbsoluteKnobSliderMappingItem:(MIKMIDIMappingItem **)mappingItem fromMessages:(NSArray *)messages
312 {
313  for (MIKMIDICommand *message in messages) {
314  if ([message respondsToSelector:@selector(isFourteenBitCommand)] &&
315  [(MIKMIDIControlChangeCommand *)message isFourteenBitCommand]) {
316  // For now, we're assuming that controllers don't do this with 14 bit commands
317  return NO;
318  }
319  }
320 
321  if (![self fillInAbsoluteKnobSliderMappingItem:mappingItem fromMessages:messages]) return NO;
322 
323  // Determine if it's a "fake" absolute knob by looking at the time between messages, velocity,
324  // and whether the value increases by exactly one each time.
325  NSTimeInterval medianTimeBetweenMessages = 0;
326  NSMutableArray *timesBetweenMessages = [NSMutableArray array];
327  double totalValueChange = 0;
328  MIKMIDIChannelVoiceCommand *previousMessage = nil;
329  for (MIKMIDIChannelVoiceCommand *message in messages) {
330  if (previousMessage) {
331  totalValueChange += fabs(MIKMIDIControlValueFromChannelVoiceCommand(message) - MIKMIDIControlValueFromChannelVoiceCommand(previousMessage));
332  [timesBetweenMessages addObject:@([message.timestamp timeIntervalSinceDate:previousMessage.timestamp])];
333  }
334  previousMessage = message;
335  }
336  [timesBetweenMessages sortUsingSelector:@selector(compare:)];
337  medianTimeBetweenMessages = [[timesBetweenMessages objectAtIndex:([timesBetweenMessages count] / 2)] doubleValue];
338  double valueChangePerMessage = totalValueChange / (double)([messages count] - 1);
339 
340  // If we see a big value change per message (>1.2) we assume it's a real absolute knob, despite
341  // other indicators. This is because some controllers throttle the messages coming from a absolute knobs so the
342  // time between messages is long, but there's a big value change for each message.
343  if (valueChangePerMessage > 1.1) return NO;
344 
345  // If the time between messages is short, it's probably not a relative absolute knob either.
346  if (medianTimeBetweenMessages < 0.02) return NO;
347 
348  [*mappingItem setInteractionType:MIKMIDIResponderTypeRelativeAbsoluteKnob];
349 
350  return YES;
351 }

353 - (MIKMIDIMappingItem *)mappingItemForCommandIdentifier:(NSString *)commandID inControl:(id<MIKMIDIMappableResponder>)responder fromReceivedMessages:(NSArray *)messages
354 {
355  if (![messages count]) return nil;
356  /* The logic here is as follows:
357 
358  For knobs and sliders:
359  We assume knobs/sliders have been moved from right-to-left or top-to-bottom, meaning increasing.
360  If the message values *decrease*, it's an indication that the control is flipped from what we expect,
361  and we need to handle that.
362 
363  If the value of each message is the same, or flips between two binary values (e.g. user twisted back and forth),
364  it's a jog wheel rather than an absolute pot.
365 
366  For buttons:
367  If we've only got one message, and it has been more than the timeout interval since then, the button is a tap type button.
368  If we've gotten two messages, with the second having value 0, the button is a key type button.
369  */
370 
371  MIKMIDIResponderType responderType = [responder MIDIResponderTypeForCommandIdentifier:commandID];
372 
373  MIKMIDIMappingItem *result = [[MIKMIDIMappingItem alloc] initWithMIDIResponderIdentifier:[responder MIDIIdentifier] andCommandIdentifier:commandID];
374 
375  if (responderType & MIKMIDIResponderTypeButton &&
376  [self fillInButtonMappingItem:&result fromMessages:messages]) {
377  goto FINALIZE_RESULT_AND_RETURN;
378  }
379 
380  if (responderType & MIKMIDIResponderTypeTurntableKnob &&
381  [self fillInTurntableKnobMappingItem:&result fromMessages:messages]) {
382  goto FINALIZE_RESULT_AND_RETURN;
383  }
384 
385  if (responderType & MIKMIDIResponderTypeRelativeKnob &&
386  [self fillInRelativeKnobMappingItem:&result fromMessages:messages]) {
387  goto FINALIZE_RESULT_AND_RETURN;
388  }
389 
390  if (responderType & MIKMIDIResponderTypeRelativeAbsoluteKnob &&
391  [self fillInRelativeAbsoluteKnobSliderMappingItem:&result fromMessages:messages]) {
392  goto FINALIZE_RESULT_AND_RETURN;
393  }
394 
395  if (responderType & MIKMIDIResponderTypeAbsoluteSliderOrKnob &&
396  [self fillInAbsoluteKnobSliderMappingItem:&result fromMessages:messages]) {
397  goto FINALIZE_RESULT_AND_RETURN;
398  }
399 
400  if (self.isDiagnosticLoggingEnabled) {
401  NSLog(@"MIDI Mapping Generator: Unable to create mapping for %@ (%@) from messages: %@", responder, commandID, messages);
402  }
403 
404  return nil;
405 
406 FINALIZE_RESULT_AND_RETURN:
407  result.commandType = [messages[0] commandType];
408 
409  if (self.isDiagnosticLoggingEnabled) {
410  NSLog(@"MIDI Mapping Generator: Created mapping item: %@ for %@ (%@) from messages: %@", result, responder, commandID, messages);
411  }
412 
413  return result;
414 }

416 - (void)timeoutTimerFired:(NSTimer *)timer
417 {
418  MIKMIDIMappingItem *mappingItem = [self mappingItemForCommandIdentifier:self.commandIdentifierBeingLearned
419  inControl:self.controlBeingLearned
420  fromReceivedMessages:self.receivedMessages];
421 
422  NSSet *existingMappingItemsForOtherControls = [self existingMappingItemsForResponderMappedTo:mappingItem];
423 
424  if (mappingItem && [existingMappingItemsForOtherControls count]) {
425  MIKMIDIMappingGeneratorRemapBehavior behavior = MIKMIDIMappingGeneratorRemapDefault;
426  id<MIKMIDIMappingGeneratorDelegate> delegate = self.delegate;
427  if ([delegate respondsToSelector:@selector(mappingGenerator:behaviorForRemappingControlMappedWithItems:toNewResponder:commandIdentifier:)]) {
428  behavior = [delegate mappingGenerator:self
429  behaviorForRemappingControlMappedWithItems:existingMappingItemsForOtherControls
430  toNewResponder:self.controlBeingLearned
431  commandIdentifier:self.commandIdentifierBeingLearned];
432  }
433 
434  switch (behavior) {
435  default:
436  case MIKMIDIMappingGeneratorRemapDisallow:
437  mappingItem = nil; // Discard this mapping item
438  break;
439  case MIKMIDIMappingGeneratorRemapAllowDuplicate:
440  // Do nothing special
441  break;
442  case MIKMIDIMappingGeneratorRemapReplace:
443  // Remove the existing mapping items.
444  [self.mapping removeMappingItems:existingMappingItemsForOtherControls];
445  break;
446  }
447  }
448 
449  if (mappingItem) {
450  [self finishMappingItem:mappingItem error:nil];
451  } else {
452  // Start over listening
453  [self.receivedMessages removeAllObjects];
454  }
455 }

457 - (void)finishMappingItem:(MIKMIDIMappingItem *)mappingItemOrNil error:(NSError *)errorOrNil
458 {
459  MIKMIDIMappingGeneratorMappingCompletionBlock completionBlock = self.currentMappingCompletionBlock;
460  id<MIKMIDIMappingGeneratorDelegate> delegate = self.delegate;
461 
462  self.currentMappingCompletionBlock = nil;
463  self.controlBeingLearned = nil;
464  NSArray *receivedMessages = [self.receivedMessages copy];
465  [self.receivedMessages removeAllObjects];
466  self.messagesTimeoutTimer = nil;
467 
468  // Determine if existing mapping items for this control should be removed.
469  BOOL shouldRemoveExisting = mappingItemOrNil != nil;
470  if (mappingItemOrNil &&
471  [self.existingMappingItems count] &&
472  [delegate respondsToSelector:@selector(mappingGenerator:shouldRemoveExistingMappingItems:forResponderBeingMapped:)]) {
473  shouldRemoveExisting = [delegate mappingGenerator:self
474  shouldRemoveExistingMappingItems:self.existingMappingItems
475  forResponderBeingMapped:self.controlBeingLearned];
476  }
477  if (shouldRemoveExisting && [self.existingMappingItems count]) [self.mapping removeMappingItems:self.existingMappingItems];
478  self.existingMappingItems = nil;
479 
480 
481  if (mappingItemOrNil) [self.mapping addMappingItemsObject:mappingItemOrNil];
482  if (completionBlock) completionBlock(mappingItemOrNil, receivedMessages, errorOrNil);
483 }

494 - (BOOL)command:(MIKMIDIChannelVoiceCommand *)command1 isSameTypeChannelNumberAsCommand:(MIKMIDIChannelVoiceCommand *)command2
495 {
496  if (command1.channel != command2.channel) return NO;
497  if (MIKMIDIControlNumberFromCommand(command1) != MIKMIDIControlNumberFromCommand(command2)) return NO;
498 
499  BOOL isDifferentCommandType = command1.commandType != command2.commandType;
500  BOOL areNoteCommands = (command1.commandType == MIKMIDICommandTypeNoteOn || command1.commandType == MIKMIDICommandTypeNoteOff) &&
501  (command2.commandType == MIKMIDICommandTypeNoteOn || command2.commandType == MIKMIDICommandTypeNoteOff);
502  isDifferentCommandType &= !areNoteCommands;
503  if (isDifferentCommandType) return NO;
504 
505  return YES;
506 }

508 - (NSSet *)existingMappingItemsForResponderMappedTo:(MIKMIDIMappingItem *)mappingItem
509 {
510  if (!mappingItem) return [NSMutableSet set];
511 
512  MIKMutableMIDIChannelVoiceCommand *matchingCommand = [MIKMutableMIDIChannelVoiceCommand commandForCommandType:mappingItem.commandType];
513  matchingCommand.channel = mappingItem.channel;
514  matchingCommand.dataByte1 = mappingItem.controlNumber;
515 
516  NSSet *existingMappingItems = [self.mapping mappingItemsForMIDICommand:matchingCommand];
517  NSMutableSet *result = [existingMappingItems mutableCopy];
518  if ([self.commandIdentifierBeingLearned length] && self.controlBeingLearned) {
519  NSSet *existingForCurrentResponder = [self.mapping mappingItemsForCommandIdentifier:self.commandIdentifierBeingLearned responder:self.controlBeingLearned];
520  [result minusSet:existingForCurrentResponder];
521  }
522  return result;
523 }

527 - (BOOL)connectToDevice:(NSError **)error
528 {
529  error = error ? error : &(NSError *__autoreleasing){ nil };
530  if (!self.device) {
531  *error = [NSError MIKMIDIErrorWithCode:MIKMIDIUnknownErrorCode userInfo:nil];
532  return NO;
533  }
534 
535  NSArray *sources = [self.device.entities valueForKeyPath:@"@unionOfArrays.sources"];
536  if (![sources count]) {
537  *error = [NSError MIKMIDIErrorWithCode:MIKMIDIDeviceHasNoSourcesErrorCode userInfo:nil];
538  return NO;
539  }
540  MIKMIDISourceEndpoint *source = [sources objectAtIndex:0];
541 
542  MIKMIDIDeviceManager *manager = [MIKMIDIDeviceManager sharedDeviceManager];
543  __weak MIKMIDIMappingGenerator *weakSelf = self;
544  id connectionToken = [manager connectInput:source error:error eventHandler:^(MIKMIDISourceEndpoint *source, NSArray *commands) {
545 
546  __strong typeof(weakSelf) strongSelf = weakSelf;
547  id<MIKMIDIMappingGeneratorDelegate> delegate = strongSelf.delegate;
548 
549  for (MIKMIDICommand *command in commands) {
550  if (![command isKindOfClass:[MIKMIDIChannelVoiceCommand class]]) continue;
551 
552  MIKMIDICommand *processedCommand = command;
553  if ([delegate respondsToSelector:@selector(mappingGenerator:commandByProcessingIncomingCommand:)]) {
554  processedCommand = [delegate mappingGenerator:self
555  commandByProcessingIncomingCommand:(MIKMIDIChannelVoiceCommand *)command];
556  if (!processedCommand) continue;
557  if (![processedCommand isKindOfClass:[MIKMIDIChannelVoiceCommand class]]) {
558  [NSException raise:NSInternalInconsistencyException format:@"-mappingGenerator:commandByProcessingCommand: must only return instances of MIKMIDIChannelVoiceCommand or one of its subclasses."];
559  continue;
560  }
561  }
562 
563  [strongSelf handleMIDICommand:(MIKMIDIChannelVoiceCommand *)processedCommand];
564  }
565  }];
566  self.connectionToken = connectionToken;
567  return connectionToken != nil;
568 }

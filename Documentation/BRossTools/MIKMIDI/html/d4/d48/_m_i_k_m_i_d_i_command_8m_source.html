55 + (NSArray *)commandsWithMIDIPacket:(MIDIPacket *)inputPacket
56 {
57  NSMutableArray *result = [NSMutableArray array];
58  NSInteger dataOffset = 0;
59  while (dataOffset < inputPacket->length) {
60  const Byte *packetData = inputPacket->data + dataOffset;
61  MIKMIDICommandType commandType = (MIKMIDICommandType)packetData[0];
62  NSInteger standardLength = MIKMIDIStandardLengthOfMessageForCommandType(commandType);
63  if (commandType == MIKMIDICommandTypeSystemExclusive) {
64  // For sysex, the packet can only contain a single MIDI message (as per documentation for MIDIPacket)
65  standardLength = inputPacket->length;
66  }
67  if (dataOffset > (inputPacket->length - standardLength)) break;
68 
69  // This is gross, but it's the only way I can find to reliably create a
70  // single-message MIDIPacket.
71  MIDIPacketList packetList;
72  MIDIPacket *midiPacket = MIDIPacketListInit(&packetList);
73  midiPacket = MIDIPacketListAdd(&packetList,
74  sizeof(MIDIPacketList),
75  midiPacket,
76  inputPacket->timeStamp,
77  standardLength,
78  packetData);
79 
80  MIKMIDICommand *command = [MIKMIDICommand commandWithMIDIPacket:midiPacket];
81  if (command) [result addObject:command];
82  dataOffset += standardLength;
83  }
84 
85  return result;
86 }

153 + (Class)subclassForCommandType:(MIKMIDICommandType)commandType
154 {
155  Class result = nil;
156  for (Class subclass in registeredMIKMIDICommandSubclasses) {
157  if ([[subclass supportedMIDICommandTypes] containsObject:@(commandType)]) {
158  result = subclass;
159  break;
160  }
161  }
162  if (!result) {
163  // Try again ignoring lower 4 bits
164  commandType |= 0x0f;
165  for (Class subclass in registeredMIKMIDICommandSubclasses) {
166  if ([[subclass supportedMIDICommandTypes] containsObject:@(commandType)]) {
167  result = subclass;
168  break;
169  }
170  }
171  }
172  return result;
173 }

337 ByteCount MIKMIDIPacketListSizeForCommands(NSArray *commands)
338 {
339  if (commands == nil || [commands count] == 0) {
340  return 0;
341  }
342 
343  // Compute the size of static members of MIDIPacketList and (MIDIPacket * [commands count])
344  ByteCount packetListSize = offsetof(MIDIPacketList, packet) + offsetof(MIDIPacket, data) * [commands count];
345 
346  // Compute the total number of MIDI bytes in all commands
347  for (MIKMIDICommand *command in commands) {
348  packetListSize += [[command data] length];
349  }
350 
351  return packetListSize;
352 }

354 BOOL MIKCreateMIDIPacketListFromCommands(MIDIPacketList **outPacketList, NSArray *commands)
355 {
356  if (outPacketList == NULL || commands == nil || [commands count] == 0) {
357  return NO;
358  }
359 
360  ByteCount listSize = MIKMIDIPacketListSizeForCommands(commands);
361 
362  if (listSize == 0) {
363  return NO;
364  }
365 
366  MIDIPacketList *packetList = calloc(1, listSize);
367  if (packetList == NULL) {
368  return NO;
369  }
370 
371  MIDIPacket *currentPacket = MIDIPacketListInit(packetList);
372  for (NSUInteger i=0; i<[commands count]; i++) {
373  MIKMIDICommand *command = [commands objectAtIndex:i];
374  currentPacket = MIDIPacketListAdd(packetList,
375  listSize,
376  currentPacket,
377  command.midiTimestamp,
378  [command.data length],
379  [command.data bytes]);
380  if (!currentPacket && (i < [commands count] - 1)) {
381  free(packetList);
382  return NO;
383  }
384  }
385 
386  *outPacketList = packetList;
387  return YES;
388 }

204 - (BOOL)sendBankSelectAndProgramChangeForInstrumentID:(MusicDeviceInstrumentID)instrumentID error:(NSError **)error
205 {
206  error = error ?: &(NSError *__autoreleasing){ nil };
207 
208  for (UInt8 channel = 0; channel < 16; channel++) {
209  // http://lists.apple.com/archives/coreaudio-api/2002/Sep/msg00015.html
210 
211 
212  CFURLRef loadedSoundfontURL = NULL;
213  UInt32 size = sizeof(loadedSoundfontURL);
214  OSStatus err = AudioUnitGetProperty(self.instrumentUnit,
215  kMusicDeviceProperty_SoundBankURL,
216  kAudioUnitScope_Global,
217  0,
218  &loadedSoundfontURL,
219  &size);
220  if (err && err != kAudioUnitErr_InvalidProperty) {
221  NSLog(@"AudioUnitGetProperty() (kMusicDeviceProperty_SoundBankURL) failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
222  *error = [NSError errorWithDomain:NSPOSIXErrorDomain code:err userInfo:nil];
223  return NO;
224  }
225 
226  if (loadedSoundfontURL || [self isUsingAppleDLSSynth]) {
227 
228  UInt32 bankSelectStatus = 0xB0 | channel;
229 
230  UInt8 bankSelectMSB = (instrumentID >> 16) & 0x7F;
231  err = _sendMIDICommand(self, self.instrumentUnit, bankSelectStatus, 0x00, bankSelectMSB, 0);
232  if (err) {
233  NSLog(@"MusicDeviceMIDIEvent() (MSB Bank Select) failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
234  *error = [NSError errorWithDomain:NSPOSIXErrorDomain code:err userInfo:nil];
235  return NO;
236  }
237 
238  UInt8 bankSelectLSB = (instrumentID >> 8) & 0x7F;
239  err = _sendMIDICommand(self, self.instrumentUnit, bankSelectStatus, 0x20, bankSelectLSB, 0);
240  if (err) {
241  NSLog(@"MusicDeviceMIDIEvent() (LSB Bank Select) failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
242  *error = [NSError errorWithDomain:NSPOSIXErrorDomain code:err userInfo:nil];
243  return NO;
244  }
245  }
246 
247  UInt32 programChangeStatus = 0xC0 | channel;
248  UInt8 programChange = instrumentID & 0x7F;
249  err = _sendMIDICommand(self, self.instrumentUnit, programChangeStatus, programChange, 0, 0);
250  if (err) {
251  NSLog(@"MusicDeviceMIDIEvent() (Program Change) failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
252  *error = [NSError errorWithDomain:NSPOSIXErrorDomain code:err userInfo:nil];
253  return NO;
254  }
255  }
256 
257  return YES;
258 }

263 {
264  error = error ?: &(NSError *__autoreleasing){ nil };
265  AUGraph graph;
266  OSStatus err = 0;
267  if ((err = NewAUGraph(&graph))) {
268  NSLog(@"Unable to create AU graph: %@", @(err));
269  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
270  return NO;
271  }
272 
273  AudioComponentDescription outputcd = {0};
274  outputcd.componentType = kAudioUnitType_Output;
275 #if TARGET_OS_IPHONE
276  outputcd.componentSubType = kAudioUnitSubType_RemoteIO;
277 #else
278  outputcd.componentSubType = kAudioUnitSubType_DefaultOutput;
279 #endif
280 
281  outputcd.componentManufacturer = kAudioUnitManufacturer_Apple;
282 
283  AUNode outputNode;
284  if ((err = AUGraphAddNode(graph, &outputcd, &outputNode))) {
285  NSLog(@"Unable to add ouptput node to graph: %@", @(err));
286  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
287  return NO;
288  }
289 
290  AudioComponentDescription instrumentcd = self.componentDescription;
291 
292  AUNode instrumentNode;
293  if ((err = AUGraphAddNode(graph, &instrumentcd, &instrumentNode))) {
294  NSLog(@"Unable to add instrument node to AU graph: %@", @(err));
295  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
296  return NO;
297  }
298 
299  if ((err = AUGraphOpen(graph))) {
300  NSLog(@"Unable to open AU graph: %@", @(err));
301  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
302  return NO;
303  }
304 
305  AudioUnit instrumentUnit;
306  if ((err = AUGraphNodeInfo(graph, instrumentNode, NULL, &instrumentUnit))) {
307  NSLog(@"Unable to get instrument AU unit: %@", @(err));
308  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
309  return NO;
310  }
311 
312  if ((err = AUGraphConnectNodeInput(graph, instrumentNode, 0, outputNode, 0))) {
313  NSLog(@"Unable to connect instrument to output: %@", @(err));
314  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
315  return NO;
316  }
317 
318  if ((err = AUGraphInitialize(graph))) {
319  NSLog(@"Unable to initialize AU graph: %@", @(err));
320  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
321  return NO;
322  }
323 
324 #if !TARGET_OS_IPHONE
325  // Turn down reverb which is way too high by default
326  if (instrumentcd.componentSubType == kAudioUnitSubType_DLSSynth) {
327  if ((err = AudioUnitSetParameter(instrumentUnit, kMusicDeviceParam_ReverbVolume, kAudioUnitScope_Global, 0, -120, 0))) {
328  NSLog(@"Unable to set reverb level to -120: %@", @(err));
329  }
330  }
331 #endif
332 
333  if ((err = AUGraphStart(graph))) {
334  NSLog(@"Unable to start AU graph: %@", @(err));
335  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
336  return NO;
337  }
338 
339  self.graph = graph;
340  self.instrumentUnit = instrumentUnit;
341 
342  return YES;
343 }

382 - (void)scheduleMIDICommands:(NSArray *)commands
383 {
384  for (MIKMIDICommand *command in commands) {
385  dispatch_sync(_scheduledCommandQueue, ^{
386  NSUInteger count = commands.count;
387  if (!self->_scheduledCommandsByTimeStamp) {
388  self->_scheduledCommandsByTimeStamp = CFDictionaryCreateMutable(NULL, count, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks);
389  }
390  if (!self->_scheduledCommandTimeStampsSet) {
391  self->_scheduledCommandTimeStampsSet = CFSetCreateMutable(NULL, count, &kCFTypeSetCallBacks);
392  }
393  if (!self->_scheduledCommandTimeStampsArray) {
394  self->_scheduledCommandTimeStampsArray = CFArrayCreateMutable(NULL, count, &kCFTypeArrayCallBacks);
395  }
396 
397  MIDITimeStamp timeStamp = command.midiTimestamp;
398  void *timeStampNumber = (__bridge void*)@(timeStamp);
399  CFMutableArrayRef commandsAtTimeStamp = (CFMutableArrayRef)CFDictionaryGetValue(self->_scheduledCommandsByTimeStamp, timeStampNumber);
400  if (!commandsAtTimeStamp) {
401  commandsAtTimeStamp = CFArrayCreateMutable(NULL, 0, &kCFTypeArrayCallBacks);
402  CFDictionarySetValue(self->_scheduledCommandsByTimeStamp, timeStampNumber, (void *)commandsAtTimeStamp);
403  CFRelease(commandsAtTimeStamp);
404 
405  if (!CFSetContainsValue(self->_scheduledCommandTimeStampsSet, timeStampNumber)) {
406  CFSetAddValue(self->_scheduledCommandTimeStampsSet, timeStampNumber);
407  CFArrayAppendValue(self->_scheduledCommandTimeStampsArray, timeStampNumber);
408  }
409  }
410 
411  CFArrayAppendValue(commandsAtTimeStamp, (__bridge void *)command);
412  });
413  }
414 }

418 OSStatus MIKMIDISynthesizerScheduleUpcomingMIDICommands(MIKMIDISynthesizer *synth, AudioUnit instrumentUnit, UInt32 inNumberFrames, Float64 sampleRate, const AudioTimeStamp *inTimeStamp)
419 {
420  dispatch_queue_t queue = synth->_scheduledCommandQueue;
421  if (!queue) return noErr; // no commands have been scheduled with this synth
422 
423  static NSTimeInterval lastTimeUntilNextCallback = 0;
424  static MIDITimeStamp lastMIDITimeStampsUntilNextCallback = 0;
425  NSTimeInterval timeUntilNextCallback = inNumberFrames / sampleRate;
426  MIDITimeStamp midiTimeStampsUntilNextCallback = lastMIDITimeStampsUntilNextCallback;
427 
428  if (lastTimeUntilNextCallback != timeUntilNextCallback) {
429  midiTimeStampsUntilNextCallback = MIKMIDIClockMIDITimeStampsPerTimeInterval(timeUntilNextCallback);
430  lastTimeUntilNextCallback = timeUntilNextCallback;
431  lastMIDITimeStampsUntilNextCallback = midiTimeStampsUntilNextCallback;
432  }
433 
434  MIDITimeStamp toTimeStamp = inTimeStamp->mHostTime + midiTimeStampsUntilNextCallback;
435  CFMutableArrayRef commandsToSend = CFArrayCreateMutable(NULL, 0, &kCFTypeArrayCallBacks);;
436 
437  dispatch_sync(queue, ^{
438  CFMutableDictionaryRef commandsByTimeStamp = synth->_scheduledCommandsByTimeStamp;
439  if (!commandsByTimeStamp || !CFDictionaryGetCount(commandsByTimeStamp)) return;
440 
441 
442  CFMutableSetRef commandTimeStampsSet = synth->_scheduledCommandTimeStampsSet;
443  CFMutableArrayRef commandTimeStampsArray = synth->_scheduledCommandTimeStampsArray;
444  if (!commandTimeStampsSet || !commandTimeStampsArray) return;
445 
446  CFArrayRef commandTimeStampsArrayCopy = CFArrayCreateCopy(NULL, commandTimeStampsArray);
447  CFIndex count = CFArrayGetCount(commandTimeStampsArrayCopy);
448  for (CFIndex i = 0; i < count; i++) {
449  NSNumber *timeStampNumber = (__bridge NSNumber *)CFArrayGetValueAtIndex(commandTimeStampsArrayCopy, i);
450  MIDITimeStamp timeStamp = timeStampNumber.unsignedLongLongValue;
451  if (timeStamp >= toTimeStamp) break;
452 
453  CFMutableArrayRef commandsAtTimeStamp = (CFMutableArrayRef)CFDictionaryGetValue(commandsByTimeStamp, (__bridge void*)timeStampNumber);
454  CFArrayAppendArray(commandsToSend, commandsAtTimeStamp, CFRangeMake(0, CFArrayGetCount(commandsAtTimeStamp)));
455 
456  CFDictionaryRemoveValue(commandsByTimeStamp, (__bridge void *)timeStampNumber);
457  CFSetRemoveValue(commandTimeStampsSet, (__bridge void*)timeStampNumber);
458  CFArrayRemoveValueAtIndex(commandTimeStampsArray, 0);
459  }
460  CFRelease(commandTimeStampsArrayCopy);
461  });
462 
463  NSTimeInterval secondsPerMIDITimeStamp = MIKMIDIClockSecondsPerMIDITimeStamp();
464 
465  CFIndex commandCount = CFArrayGetCount(commandsToSend);
466  for (CFIndex i = 0; i < commandCount; i++) {
467  MIKMIDICommand *command = (__bridge MIKMIDICommand *)CFArrayGetValueAtIndex(commandsToSend, i);
468 
469  MIDITimeStamp sendTimeStamp = command.midiTimestamp;
470  if (sendTimeStamp < inTimeStamp->mHostTime) sendTimeStamp = inTimeStamp->mHostTime;
471  MIDITimeStamp timeStampOffset = sendTimeStamp - inTimeStamp->mHostTime;
472  Float64 sampleOffset = secondsPerMIDITimeStamp * timeStampOffset * sampleRate;
473 
474  OSStatus err = synth->_sendMIDICommand(synth, instrumentUnit, command.statusByte, command.dataByte1, command.dataByte2, sampleOffset);
475  if (err) {
476  NSLog(@"Unable to schedule MIDI command %@ for instrument unit %p: %@", command, instrumentUnit, @(err));
477  return err;
478  }
479  }
480 
481  CFRelease(commandsToSend);
482  return noErr;
483 }

485 static OSStatus MIKMIDISynthesizerInstrumentUnitRenderCallback(void * inRefCon,
486  AudioUnitRenderActionFlags * ioActionFlags,
487  const AudioTimeStamp * inTimeStamp,
488  UInt32 inBusNumber,
489  UInt32 inNumberFrames,
490  AudioBufferList * ioData)
491 {
492  if (*ioActionFlags & kAudioUnitRenderAction_PreRender) {
493  if (!(inTimeStamp->mFlags & kAudioTimeStampHostTimeValid)) return noErr;
494  if (!(inTimeStamp->mFlags & kAudioTimeStampSampleTimeValid)) return noErr;
495 
496  MIKMIDISynthesizer *synth = (__bridge MIKMIDISynthesizer *)inRefCon;
497  AudioUnit instrumentUnit = synth.instrumentUnit;
498  AudioStreamBasicDescription LPCMASBD;
499  UInt32 sizeOfLPCMASBD = sizeof(LPCMASBD);
500  OSStatus err = AudioUnitGetProperty(instrumentUnit, kAudioUnitProperty_StreamFormat, kAudioUnitScope_Output, 0, &LPCMASBD, &sizeOfLPCMASBD);
501  if (err) {
502  NSLog(@"Unable to get stream description for instrument unit %p: %@", instrumentUnit, @(err));
503  return err;
504  }
505 
506  return MIKMIDISynthesizerScheduleUpcomingMIDICommands(synth, instrumentUnit, inNumberFrames, LPCMASBD.mSampleRate, inTimeStamp);
507  }
508  return noErr;
509 }

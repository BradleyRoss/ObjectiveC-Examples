103 - (void)retrieveEntities
104 {
105  NSMutableArray *entities = [NSMutableArray array];
106 
107  ItemCount numEntities = MIDIDeviceGetNumberOfEntities(self.objectRef);
108  for (ItemCount i=0; i<numEntities; i++) {
109  MIDIEntityRef entityRef = MIDIDeviceGetEntity(self.objectRef, i);
110  MIKMIDIEntity *entity = [MIKMIDIEntity MIDIObjectWithObjectRef:entityRef];
111  if (!entity) continue;
112  entity.device = self;
113  [entities addObject:entity];
114  }
115 
116  self.internalEntities = entities;
117 }

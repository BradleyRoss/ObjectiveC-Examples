66 MIDIObjectType MIKMIDIObjectTypeOfObject(MIDIObjectRef object, NSError *__autoreleasing*error)
67 {
68  error = error ? error : &(NSError *__autoreleasing){ nil };
69  MIDIUniqueID uniqueID = MIKIntegerPropertyFromMIDIObject(object, kMIDIPropertyUniqueID, error);
70  if (*error) return -2;
71 
72  MIDIObjectRef resultObject;
73  MIDIObjectType objectType;
74  OSStatus err = MIDIObjectFindByUniqueID(uniqueID, &resultObject, &objectType);

76  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
77  return -2;
78  }
79 
80  if (resultObject != object) {

82  return -2;
83  }
84 
85  return objectType;
86 }

110 NSInteger _MIKMIDIStandardLengthOfMessageForCommandType(MIKMIDICommandType commandType)
111 {
112  // Result includes status/command type byte
113  switch (commandType) {
114  case MIKMIDICommandTypeNoteOff:
115  case MIKMIDICommandTypeNoteOn:
116  case MIKMIDICommandTypePolyphonicKeyPressure:
117  case MIKMIDICommandTypeControlChange:
118  case MIKMIDICommandTypePitchWheelChange:
119  case MIKMIDICommandTypeSystemSongPositionPointer:
120  return 3;
121  break;
122  case MIKMIDICommandTypeProgramChange:
123  case MIKMIDICommandTypeChannelPressure:
124  case MIKMIDICommandTypeSystemTimecodeQuarterFrame:
125  case MIKMIDICommandTypeSystemSongSelect:
126  return 2;
127  break;
128  case MIKMIDICommandTypeSystemTuneRequest:
129  case MIKMIDICommandTypeSystemTimingClock:
130  case MIKMIDICommandTypeSystemStartSequence:
131  case MIKMIDICommandTypeSystemContinueSequence:
132  case MIKMIDICommandTypeSystemStopSequence:
133  case MIKMIDICommandTypeSystemKeepAlive:
134  return 1;
135  break;
136  case MIKMIDICommandTypeSystemMessage:
137  case MIKMIDICommandTypeSystemExclusive:
138  return -1; // No standard length
139  break;
140  default:
141  return NSIntegerMin;
142  break;
143  }
144 }

158 MIDIPacket MIKMIDIPacketCreate(MIDITimeStamp timeStamp, UInt16 length, MIKArrayOf(NSNumber *) *data /*max length 256*/)
159 {
160  MIDIPacket result = {0};
161  if ([data count] > 256) {
162  [NSException raise:NSInvalidArgumentException format:@"MIKMIDIPacketCreate()'s data argument must contain 256 or fewer values"];
163  return result;
164  }
165 
166  result.timeStamp = timeStamp;
167  result.length = length;
168  for (NSUInteger i=0; i<256; i++) {
169  if (i >= [data count]) {
170  result.data[i] = 0;
171  continue;
172  }
173 
174  result.data[i] = [data[i] charValue];
175  }
176 
177  return result;
178 }

180 MIDIPacket *MIKMIDIPacketCreateFromCommands(MIDITimeStamp timeStamp, MIKArrayOf(MIKMIDICommand *) *commands)
181 {
182  NSMutableData *allPacketData = [NSMutableData data];
183  for (MIKMIDICommand *command in commands) {
184  [allPacketData appendData:command.data];
185  }
186 
187  MIDIPacket *result = malloc(sizeof(MIDIPacket) + allPacketData.length);
188  result->timeStamp = timeStamp;
189  result->length = allPacketData.length;
190  [allPacketData getBytes:result->data length:allPacketData.length];
191 
192  return result;

203 {
204  NSUInteger scaledNoteNumber = noteNumber % 12;

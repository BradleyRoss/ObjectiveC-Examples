95 {

108 - (void)syncMusicTimeStamp:(MusicTimeStamp)musicTimeStamp withMIDITimeStamp:(MIDITimeStamp)midiTimeStamp tempo:(Float64)tempo
109 {
110  [self willChangeValueForKey:@"ready"];
111  dispatchToClockQueue(self, ^{
112  if (self->_lastSyncedMIDITimeStamp != 0) {
113  // Add a clock to the historical clocks
114  NSNumber *midiTimeStampNumber = @(midiTimeStamp);
115 
116  if (!self->_historicalClocks) {
117  self->_historicalClocks = CFDictionaryCreateMutable(NULL, 0, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks);
118  self->_historicalClockMIDITimeStampsSet = CFSetCreateMutable(NULL, 0, &kCFTypeSetCallBacks);
119  self->_historicalClockMIDITimeStampsArray = CFArrayCreateMutable(NULL, 0, &kCFTypeArrayCallBacks);
120  } else {
121  // Remove clocks old enough to not be needed anymore
122  MIDITimeStamp oldTimeStamp = MIKMIDIGetCurrentTimeStamp() - MIKMIDIClockMIDITimeStampsPerTimeInterval(kDurationToKeepHistoricalClocks);
123 
124  CFIndex count = CFArrayGetCount(self->_historicalClockMIDITimeStampsArray);
125  CFMutableArrayRef timeStampsToRemove = CFArrayCreateMutable(NULL, 0, &kCFTypeArrayCallBacks);
126  CFMutableSetRef indexesToRemoveSet = CFSetCreateMutable(NULL, 0, &kCFTypeSetCallBacks);
127  CFMutableArrayRef indexesToRemoveArray = CFArrayCreateMutable(NULL, 0, &kCFTypeArrayCallBacks);
128 
129  for (CFIndex i = 0; i < count; i++) {
130  NSNumber *timeStampNumber = (__bridge NSNumber *)CFArrayGetValueAtIndex(self->_historicalClockMIDITimeStampsArray, i);
131  MIDITimeStamp timeStamp = timeStampNumber.unsignedLongLongValue;
132  if (timeStamp <= oldTimeStamp) {
133  void *timeStampValue = (__bridge void *)timeStampNumber;
134 
135  CFArrayAppendValue(timeStampsToRemove, timeStampValue);
136  if (!CFSetContainsValue(indexesToRemoveSet, timeStampValue)) {
137  CFSetAddValue(indexesToRemoveSet, timeStampValue);
138  CFArrayAppendValue(indexesToRemoveArray, timeStampValue);
139  }
140  } else {
141  break;
142  }
143  }
144 
145  CFIndex timeStampsToRemoveCount = CFArrayGetCount(timeStampsToRemove);
146  for (CFIndex i = (timeStampsToRemoveCount - 1); i >= 0; i--) {
147  const void *timeStampValue = CFArrayGetValueAtIndex(timeStampsToRemove, i);
148  CFDictionaryRemoveValue(self->_historicalClocks, timeStampValue);
149  CFSetRemoveValue(self->_historicalClockMIDITimeStampsSet, timeStampValue);
150  CFArrayRemoveValueAtIndex(self->_historicalClockMIDITimeStampsArray, i);
151  }
152 
153  CFRelease(timeStampsToRemove);
154  CFRelease(indexesToRemoveSet);
155  CFRelease(indexesToRemoveArray);
156  }
157 
158  // Add clock to history
159  MIKMIDIClock *historicalClock = [[MIKMIDIClock alloc] initWithQueue:NO];
160  historicalClock->_currentTempo = self->_currentTempo;
161  historicalClock->_timeStampZero = self->_timeStampZero;
162  historicalClock->_lastSyncedMIDITimeStamp = self->_lastSyncedMIDITimeStamp;
163  historicalClock->_lastSyncedMusicTimeStamp = self->_lastSyncedMusicTimeStamp;
164  historicalClock->_musicTimeStampsPerMIDITimeStamp = self->_musicTimeStampsPerMIDITimeStamp;
165  historicalClock->_midiTimeStampsPerMusicTimeStamp = self->_midiTimeStampsPerMusicTimeStamp;
166 
167  void *midiTimeStampValue = (__bridge void *)midiTimeStampNumber;
168  CFDictionaryAddValue(self->_historicalClocks, midiTimeStampValue, (__bridge void *)historicalClock);
169 
170  if (!CFSetContainsValue(self->_historicalClockMIDITimeStampsSet, midiTimeStampValue)) {
171  CFSetAddValue(self->_historicalClockMIDITimeStampsSet, midiTimeStampValue);
172  CFArrayAppendValue(self->_historicalClockMIDITimeStampsArray, midiTimeStampValue);
173  }
174  }
175 
176  // Update new tempo and timing information
177  Float64 secondsPerMIDITimeStamp = MIKMIDIClockSecondsPerMIDITimeStamp();
178  Float64 secondsPerMusicTimeStamp = 60.0 / tempo;
179  Float64 midiTimeStampsPerMusicTimeStamp = secondsPerMusicTimeStamp / secondsPerMIDITimeStamp;
180 
181  self->_currentTempo = tempo;
182  self->_lastSyncedMIDITimeStamp = midiTimeStamp;
183  self->_lastSyncedMusicTimeStamp = musicTimeStamp;
184  self->_timeStampZero = midiTimeStamp - (musicTimeStamp * midiTimeStampsPerMusicTimeStamp);
185  self->_midiTimeStampsPerMusicTimeStamp = midiTimeStampsPerMusicTimeStamp;
186  self->_musicTimeStampsPerMIDITimeStamp = secondsPerMIDITimeStamp / secondsPerMusicTimeStamp;
187  self->_ready = YES;
188  });
189  [self didChangeValueForKey:@"ready"];
190 }

204 static MusicTimeStamp musicTimeStampForMIDITimeStamp(MIKMIDIClock *self, MIDITimeStamp midiTimeStamp)
205 {
206  __block MusicTimeStamp musicTimeStamp = 0;
207 
208  dispatchToClockQueue(self, ^{
209  if (!self->_ready) return;
210 
211  MIDITimeStamp lastSyncedMIDITimeStamp = self->_lastSyncedMIDITimeStamp;
212  if (midiTimeStamp >= lastSyncedMIDITimeStamp) {
213  musicTimeStamp = musicTimeStampForMIDITimeStampWithHistoricalClock(midiTimeStamp, self);
214  } else {
215  musicTimeStamp = musicTimeStampForMIDITimeStampWithHistoricalClock(midiTimeStamp, clockForMIDITimeStamp(self, midiTimeStamp));
216  }
217  });
218 
219  return musicTimeStamp;
220 }

227 static MusicTimeStamp musicTimeStampForMIDITimeStampWithHistoricalClock(MIDITimeStamp midiTimeStamp, MIKMIDIClock *clock)
228 {
229  if (midiTimeStamp == clock->_lastSyncedMIDITimeStamp) return clock->_lastSyncedMusicTimeStamp;
230  MIDITimeStamp timeStampZero = clock->_timeStampZero;
231  return (midiTimeStamp >= timeStampZero) ? ((midiTimeStamp - timeStampZero) * clock->_musicTimeStampsPerMIDITimeStamp) : -((timeStampZero - midiTimeStamp) * clock->_musicTimeStampsPerMIDITimeStamp);
232 }

234 static MIDITimeStamp midiTimeStampForMusicTimeStamp(MIKMIDIClock *self, MusicTimeStamp musicTimeStamp)
235 {
236  __block MIDITimeStamp midiTimeStamp = 0;
237 
238  dispatchToClockQueue(self, ^{
239  if (!self->_ready) return;
240  if (musicTimeStamp == self->_lastSyncedMusicTimeStamp) { midiTimeStamp = self->_lastSyncedMIDITimeStamp; return; }
241 
242  midiTimeStamp = round(musicTimeStamp * self->_midiTimeStampsPerMusicTimeStamp) + self->_timeStampZero;
243 
244  if (midiTimeStamp < self->_lastSyncedMIDITimeStamp && self->_historicalClockMIDITimeStampsArray) {
245  CFIndex historicalClockMIDITimeStampsCount = CFArrayGetCount(self->_historicalClockMIDITimeStampsArray);
246  for (CFIndex i = (historicalClockMIDITimeStampsCount - 1); i >= 0; i--) {
247  const void *midiTimeStampValue = CFArrayGetValueAtIndex(self->_historicalClockMIDITimeStampsArray, i);
248 
249  MIKMIDIClock *clock = (__bridge MIKMIDIClock *)CFDictionaryGetValue(self->_historicalClocks, midiTimeStampValue);
250  MIDITimeStamp historicalMIDITimeStamp = round(musicTimeStamp * clock->_midiTimeStampsPerMusicTimeStamp) + clock->_timeStampZero;
251  if (historicalMIDITimeStamp >= clock->_lastSyncedMIDITimeStamp) {
252  midiTimeStamp = historicalMIDITimeStamp;
253  break;
254  }
255  }
256  }
257  });
258 
259  return midiTimeStamp;
260 }

319 static MIKMIDIClock *clockForMIDITimeStamp(MIKMIDIClock *self, MIDITimeStamp midiTimeStamp)
320 {
321  MIKMIDIClock *clock = self;
322 
323  if (self->_historicalClockMIDITimeStampsArray) {
324  CFIndex count = CFArrayGetCount(self->_historicalClockMIDITimeStampsArray);
325  for (CFIndex i = (count - 1); i >= 0; i--) {
326  NSNumber *historicalClockTimeStamp = (__bridge NSNumber *)CFArrayGetValueAtIndex(self->_historicalClockMIDITimeStampsArray, i);
327  if ([historicalClockTimeStamp unsignedLongLongValue] > midiTimeStamp) {
328  clock = (__bridge MIKMIDIClock *)CFDictionaryGetValue(self->_historicalClocks, (__bridge void *)historicalClockTimeStamp);
329  } else {
330  break;
331  }
332  }
333  }
334 
335  return clock;
336 }

444  [invocation getArgument:&musicTimeStamp atIndex:2];
445 
446  MIDITimeStamp midiTimeStamps = midiTimeStampsPerMusicTimeStamp(_masterClock, musicTimeStamp);
447  return [invocation setReturnValue:&midiTimeStamps];

449  MIKMIDISyncedClockProxy *syncedClock = self;

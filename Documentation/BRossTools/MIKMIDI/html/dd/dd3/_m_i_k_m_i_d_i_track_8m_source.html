172 - (BOOL)insertMIDIEventInMusicTrack:(MIKMIDIEvent *)event error:(NSError **)error
173 {
174  error = error ? error : &(NSError *__autoreleasing){ nil };
175 
176  OSStatus err = noErr;
177  MusicTrack track = self.musicTrack;
178  MusicTimeStamp timeStamp = event.timeStamp;
179  const void *data = [event.data bytes];
180 
181  switch (event.eventType) {
182  case MIKMIDIEventTypeNULL:
183  NSLog(@"Warning: %s attempted to insert NULL event.", __PRETTY_FUNCTION__);
184  break;
185 
186  case MIKMIDIEventTypeExtendedNote:
187  err = MusicTrackNewExtendedNoteEvent(track, timeStamp, data);
188  if (err) NSLog(@"MusicTrackNewExtendedNoteEvent() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
189  break;
190 
191 #if !TARGET_OS_IPHONE // Unavailable altogether on iOS.
192  case MIKMIDIEventTypeExtendedControl:
193  NSLog(@"Events of type MIKMIDIEventTypeExtendedControl are unsupported because the underlying CoreMIDI API is deprecated.");
194  break;
195 #endif
196 
197  case MIKMIDIEventTypeExtendedTempo:
198  err = MusicTrackNewExtendedTempoEvent(track, timeStamp, ((ExtendedTempoEvent *)data)->bpm);
199  if (err) NSLog(@"MusicTrackNewExtendedTempoEvent() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
200  break;
201 
202  case MIKMIDIEventTypeUser:
203  err = MusicTrackNewUserEvent(track, timeStamp, data);
204  if (err) NSLog(@"MusicTrackNewUserEvent() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
205  break;
206 
207  case MIKMIDIEventTypeMIDINoteMessage:
208  err = MusicTrackNewMIDINoteEvent(track, timeStamp, data);
209  if (err) NSLog(@"MusicTrackNewMIDINoteEvent() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
210  break;
211 
212  case MIKMIDIEventTypeMIDIRawData:
213  err = MusicTrackNewMIDIRawDataEvent(track, timeStamp, data);
214  if (err) NSLog(@"MusicTrackNewMIDIRawDataEvent() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
215  break;
216 
217  case MIKMIDIEventTypeParameter:
218  err = MusicTrackNewParameterEvent(track, timeStamp, data);
219  if (err) NSLog(@"MusicTrackNewParameterEvent() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
220  break;
221 
222  case MIKMIDIEventTypeAUPreset:
223  err = MusicTrackNewAUPresetEvent(track, timeStamp, data);
224  if (err) NSLog(@"MusicTrackNewAUPresetEvent() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
225  break;
226 
227  case MIKMIDIEventTypeMIDIChannelMessage:
228  case MIKMIDIEventTypeMIDIPolyphonicKeyPressureMessage:
229  case MIKMIDIEventTypeMIDIControlChangeMessage:
230  case MIKMIDIEventTypeMIDIProgramChangeMessage:
231  case MIKMIDIEventTypeMIDIChannelPressureMessage:
232  case MIKMIDIEventTypeMIDIPitchBendChangeMessage:
233  err = MusicTrackNewMIDIChannelEvent(track, timeStamp, data);
234  if (err) NSLog(@"MusicTrackNewMIDIChannelEvent() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
235  break;
236 
237  case MIKMIDIEventTypeMeta:
238  case MIKMIDIEventTypeMetaSequence:
239  case MIKMIDIEventTypeMetaText:
240  case MIKMIDIEventTypeMetaCopyright:
241  case MIKMIDIEventTypeMetaTrackSequenceName:
242  case MIKMIDIEventTypeMetaInstrumentName:
243  case MIKMIDIEventTypeMetaLyricText:
244  case MIKMIDIEventTypeMetaMarkerText:
245  case MIKMIDIEventTypeMetaCuePoint:
246  case MIKMIDIEventTypeMetaMIDIChannelPrefix:
247  case MIKMIDIEventTypeMetaEndOfTrack:
248  case MIKMIDIEventTypeMetaTempoSetting:
249  case MIKMIDIEventTypeMetaSMPTEOffset:
250  case MIKMIDIEventTypeMetaTimeSignature:
251  case MIKMIDIEventTypeMetaKeySignature:
252  case MIKMIDIEventTypeMetaSequenceSpecificEvent:
253  err = MusicTrackNewMetaEvent(track, timeStamp, data);
254  if (err) NSLog(@"MusicTrackNewMetaEvent() failed with error %@ in %s.", @(err), __PRETTY_FUNCTION__);
255  break;
256  default:
257  err = -1;
258  NSLog(@"Warning: %s attempted to insert unknown event type %@.", __PRETTY_FUNCTION__, @(event.eventType));
259  break;
260 }
261 
262  if (err != noErr) {
263  *error = [NSError errorWithDomain:NSOSStatusErrorDomain code:err userInfo:nil];
264  return NO;
265  }
266  return YES;
267 }

269 - (BOOL)removeMIDIEventsFromMusicTrack:(NSSet *)events error:(NSError **)error
270 {
271  error = error ? error : &(NSError *__autoreleasing){ nil };
272  if (![events count]) return YES;
273 
274  // MusicTrackClear() doesn't reliably clear events that fall on its boundaries,
275  // so we iterate the track and delete that way instead
276  BOOL success = NO;
277  MIKMIDIEventIterator *iterator = [MIKMIDIEventIterator iteratorForTrack:self];
278  while (iterator.hasCurrentEvent) {
279  MIKMIDIEvent *currentEvent = iterator.currentEvent;
280  if ([events containsObject:currentEvent]) {
281  if (![iterator deleteCurrentEventWithError:error]) return NO;
282  success = YES;
283  continue; // Move to next event is done by delete.
284  }
285 
286  [iterator moveToNextEvent];
287  }
288 
289  *error = [NSError MIKMIDIErrorWithCode:MIKMIDITrackEventNotFoundErrorCode userInfo:nil];
290  return success;
291 }

323 {
324  MIKMIDIEventIterator *iterator = [MIKMIDIEventIterator iteratorForTrack:self];
325  NSMutableSet *allEvents = [NSMutableSet set];
326  while (iterator.hasCurrentEvent) {
327  MIKMIDIEvent *event = iterator.currentEvent;
328  [allEvents addObject:event];
329  [iterator moveToNextEvent];
330  }
331 
332  [self willChangeValueForKey:@"internalEvents"];

353 - (BOOL)private_moveEventsFromStartingTimeStamp:(MusicTimeStamp)startTimeStamp toEndingTimeStamp:(MusicTimeStamp)endTimeStamp byAmount:(MusicTimeStamp)timestampOffset
354 {
355  // MusicTrackMoveEvents() fails in common edge cases, so iterate the track and move that way instead
356 
357  if (timestampOffset == 0) return YES; // Nothing needs to be done
358  MusicTimeStamp length = self.length;
359  if (!length || (startTimeStamp > length) || ![self.internalEvents count]) return YES;
360  if (endTimeStamp > length) endTimeStamp = length;
361 
362  NSMutableSet *eventsToMove = [NSMutableSet setWithArray:[self eventsFromTimeStamp:startTimeStamp toTimeStamp:endTimeStamp]];
363  NSSet *eventsBeforeMoving = [eventsToMove copy];
364  NSMutableSet *eventsAfterMoving = [NSMutableSet set];
365 
366  MIKMIDIEventIterator *iterator = [MIKMIDIEventIterator iteratorForTrack:self];
367  while (iterator.hasCurrentEvent && [eventsToMove count] > 0) {
368  MIKMIDIEvent *currentEvent = iterator.currentEvent;
369  if (![eventsToMove containsObject:currentEvent]) {
370  [iterator moveToNextEvent];
371  continue;
372  }
373 
374  MusicTimeStamp timestamp = currentEvent.timeStamp;
375  if (![iterator moveCurrentEventTo:timestamp+timestampOffset error:NULL]) {
376  [self reloadAllEventsFromMusicTrack];
377  return NO;
378  }
379  MIKMutableMIDIEvent *movedEvent = [currentEvent mutableCopy];
380  movedEvent.timeStamp += timestampOffset;
381  [eventsAfterMoving addObject:[movedEvent copy]];
382  [eventsToMove removeObject:currentEvent];
383  [iterator seek:timestamp]; // Move back to previous position
384  }
385 
386  [self removeInternalEvents:eventsBeforeMoving];
387  [self addInternalEvents:eventsAfterMoving];
388 
389  return YES;
390 }

412 {
413  __block BOOL success = NO;
414 
415  [self dispatchSyncToSequencerProcessingQueueAsNeeded:^{
416  MusicTimeStamp length = self.length;
417  if (!length || (startTimeStamp > length) || ![self.internalEvents count]) { success = YES; return; }
418

457 - (BOOL)private_mergeEventsFromMIDITrack:(MIKMIDITrack *)origTrack fromTimeStamp:(MusicTimeStamp)startTimeStamp toTimeStamp:(MusicTimeStamp)endTimeStamp atTimeStamp:(MusicTimeStamp)destTimeStamp
458 {
459  NSArray *sourceEvents = [origTrack eventsFromTimeStamp:startTimeStamp toTimeStamp:endTimeStamp];
460  if (![sourceEvents count]) return YES;
461 
462  MusicTimeStamp firstSourceTimeStamp = [[sourceEvents firstObject] timeStamp];
463 
464  NSMutableSet *destinationEvents = [NSMutableSet set];
465  for (MIKMIDIEvent *event in sourceEvents) {
466  MIKMutableMIDIEvent *mutableEvent = [event mutableCopy];
467  mutableEvent.timeStamp = destTimeStamp + (event.timeStamp - firstSourceTimeStamp);
468  if (![self insertMIDIEventInMusicTrack:mutableEvent error:NULL]) {
469  [self reloadAllEventsFromMusicTrack];
470  return NO;
471  }
472  [destinationEvents addObject:mutableEvent];
473  }
474 
475  [self addInternalEvents:destinationEvents];
476  return YES;
477 }

22 /**
23  * Overview
24  * --------
25  *
26  * MIKMIDIMapping includes represents a complete mapping between a particular hardware controller,
27  * and an application's functionality. Primarily, it acts as a container for MIKMIDIMappingItems,
28  * each of which specifies the mapping for a single hardware control.
29  *
30  * MIKMIDIMapping can be stored on disk using a straightforward XML format, and includes methods
31  * to load and write these XML files. Currently this is only implemented on OS X (see
32  * https://github.com/mixedinkey-opensource/MIKMIDI/issues/2 ).
33  *
34  * Another class, MIKMIDIMappingManager can be used to manage both application-supplied, and
35  * user customized mappings.
36  *
37  * Creating Mappings
38  * -----------------
39  *
40  * MIKMIDIMappings can be generated manually, as the XML format is fairly straightforward.
41  * MIKMIDI also includes powerful functionality to assist in creating a way for users to
42  * easily create their own mappings using a "MIDI learning" interface.
43  *
44  * Using Mappings
45  * --------------
46  *
47  * MIKMIDI does not include built in support for automatically routing messages using a mapping,
48  * so a user of MIKMIDI must write some code to make this happen. Typically, this is done by having
49  * a single controller in the application be responsible for receiving all incoming MIDI messages.
50  * When a MIDI message is received, it can query the MIDI mapping for the mapping item correspoding
51  * to the incomding message, then send the command to the mapped responder. Example code for this scenario:
52  *
53  * - (void)connectToMIDIDevice:(MIKMIDIDevice *)device {
54  * MIKMIDIDeviceManager *manager = [MIKMIDIDeviceManager sharedDeviceManager];
55  * BOOL success = [manager connectInput:source error:error eventHandler:^(MIKMIDISourceEndpoint *source, NSArray *commands) {
56  * for (MIKMIDICommand *command in commands) {
57  * [self routeIncomingMIDICommand:command];
58  * }
59  * }];
60  *
61  * if (success) self.device = device;
62  * }
63  *
64  * - (void)routeIncomingMIDICommand:
65  * {
66  * MIKMIDIDevice *controller = self.device; // The connected MIKMIDIDevice instance
67  * MIKMIDIMapping *mapping = [[[MIKMIDIMappingManager sharedManager] mappingsForControllerName:controller.name] anyObject];
68  * MIKMIDIMappingItem *mappingItem = [[self.MIDIMapping mappingItemsForMIDICommand:command] anyObject];
69  * if (!mappingItem) return;
70  *
71  * id<MIKMIDIResponder> responder = [NSApp MIDIResponderWithIdentifier:mappingItem.MIDIResponderIdentifier];
72  * if ([responder respondsToMIDICommand:command]) {
73  * [responder handleMIDICommand:command];
74  * }
75  * }
76  *
77  * @see MIKMIDIMappingManager
78  * @see MIKMIDIMappingGenerator
79  */

21 /**
22  * A MIDI System Exclusive (SysEx) message. System exclusive messages are
23  * messages defined by individual manufacturers of MIDI devices. They
24  * can contain arbitrary data and can be used to support commands and responses
25  * not explicitly supported by the standard portion of MIDI spec. There are also
26  * some "Universal Exclusive Mesages", which while a type of SysEx message,
27  * are not manufacturer/device specific.
28  */

49 /**
50  * The manufacturer ID for the command. This is used by devices to determine
51  * if the message is one they support. If it is not, the message is ignored.
52  * Manufacturer IDs are assigned by the MIDI Manufacturer's Association, and
53  * a list can be found here: http://www.midi.org/techspecs/manid.php
54  *
55  * The default is 0x7E (kMIKMIDISysexNonRealtimeManufacturerID).
56  *
57  * The manufacturer ID can be either 1 byte or 3 bytes.
58  *
59  * Values 0x7E (kMIKMIDISysexNonRealtimeManufacturerID) and 0x7F (kMIKMIDISysexRealtimeManufacturerID)
60  * mean that the message is a universal (non-manufacturer specific)
61  * system exclusive message.
62  */
63 @property (nonatomic, readonly) UInt32 manufacturerID;

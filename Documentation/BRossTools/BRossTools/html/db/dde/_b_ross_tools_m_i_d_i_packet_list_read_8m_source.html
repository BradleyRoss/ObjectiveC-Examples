82 /**
83  @brief Process contents of a packet
84  @todo insert code for SysEx
85  @param packet Address of MIDIPacket

87 - (void) processPacket:(MIDIPacket *) packet {
88  NSString *message =[[NSString alloc] initWithFormat:@"Start of packet with %d bytes for %f seconds\n", packet->length, getSeconds(packet->timeStamp - startTicks)];
89  [textWindow appendString:message];
90  time=packet->timeStamp;
91  packetBytes = packet->length;
92  commandStart = 0;
93 
94  while (YES) {
95  UInt8 command = packet->data[commandStart]/16;
96  UInt8 channel = packet->data[commandStart] % 16;
97  int length;
98  switch (command) {
99  case kMIDICVStatusNoteOff: // 8
100  length = 3;
101  message = [[NSString alloc] initWithFormat:@"NoteOff ch %d p %d v %d\n", channel, packet->data[commandStart +1],packet->data[commandStart+2]];
102  break;
103  case kMIDICVStatusNoteOn: // 9
104  length = 3;
105  message = [[NSString alloc] initWithFormat:@"NoteOn ch %d p %d v %d\n", channel, packet->data[commandStart +1],packet->data[commandStart+2]];
106  break;
107  case kMIDICVStatusPolyPressure: // 10
108  message = [[NSString alloc] initWithFormat:@"PolyPressure ch %d p %d press %d\n", channel, packet->data[commandStart +1],packet->data[commandStart+2]];
109  length = 3;
110  break;
111  case kMIDICVStatusControlChange: // 11
112  message = [[NSString alloc] initWithFormat:@"ControlChange ch %d control %d val %d\n", channel, packet->data[commandStart +1],packet->data[commandStart+2]];
113  length = 3;
114  break;
115  case kMIDICVStatusProgramChange: // 12
116  message = [[NSString alloc] initWithFormat:@"ProgramChange ch %d program %d\n", channel, packet->data[commandStart +1]];
117  length = 2;
118  break;
119  case kMIDICVStatusChannelPressure: // 13
120  length = 2;
121  message = [[NSString alloc] initWithFormat:@"ChannelPressure ch %d program %d\n", channel, packet->data[commandStart +1]];
122  break;
123  case kMIDICVStatusPitchBend: // 14
124  length = 3;
125  message = [[NSString alloc] initWithFormat:@"PitchBend ch %d data %d %d\n", channel, packet->data[commandStart +1], packet->data[commandStart+2]];
126  break;
127  case 15:
128  /*
129  * @todo Needs more work for status bytes from 0xF0 to 0xFF
130  * @see https://ccrma.stanford.edu/~craig/articles/linuxmidi/misc/essenmidi.html
131  */
132  if (channel == 0) {
133  length = packetBytes + 1000;
134  message = @"MIDI System Exclusive message (start)\n";
135  } else if (channel == 1) {
136  length = 2;
137  message = @"MIDI Quarter Frame message";
138  }else if (channel == 2) {
139  length = 3;
140  message = @"MIDI Song Position message";
141  } else if (channel ==3) {
142  length = 2;
143  message = @"MIDI Song Request message\n";
144  } else if (channel == 4 || channel == 5 || channel == 9) {
145  length = packetBytes + 1000;
146  message = @"Unknown command\n";
147  } else if (channel == 6) {
148  length = 1;
149  message = @"MIDI Tune Request message";
150  } else if (channel == 7) {
151  length = packetBytes + 1000;
152  message = @"SysEx Continuation";
153  } else if (channel == 8) {
154  length = 1;
155  message = @"MIDI Clock message";
156  } else if (channel == 10) {
157  length = 1;
158  message = @"MIDI Start message\n";
159  } else if (channel == 11) {
160  length = 1;
161  message = @"MIDI Continue message\n";
162  } else if (channel == 12) {
163  length = 1;
164  message = @"MIDI Stop message\n";
165  } else if (channel == 13) {
166  length = packetBytes + 1000;
167  message = [[NSString alloc] initWithFormat:@"Unknown command - channel nibble is %xd \n", channel];
168  } else if (channel == 14) {
169  length = 1;
170  message = @"MIDI Active Sense message";
171  } else if (channel == 15) {
172  length = 1;
173  message = @"MIDI Reset message\n";
174  } else {
175  NSLog(@"This point should never be reached - aborting");
176  length = 1;
177  }
178  break;
179  default:
180  NSLog(@"Unrecognized command code %d - Should never be reached\n",command);
181  length = packetBytes + 1000;
182  }
183  printMessage(message, textWindow);
184  commandStart = commandStart + length;
185  if (commandStart >= packetBytes) {
186  return;
187  }
188  }

430  if (numer == 0) {

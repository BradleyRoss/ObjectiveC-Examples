17 /**
18  * MIKMIDISynthesizer provides a simple way to synthesize MIDI messages to
19  * produce sound output.
20  *
21  * To use it, simply create a synthesizer instance, then pass MIDI messages
22  * to it by calling -handleMIDIMessages:.
23  *
24  * A subclass, MIKMIDIEndpointSynthesizer, adds the ability to easily connect
25  * to a MIDI endpoint and automatically synthesize incoming messages.
26  *
27  * @see MIKMIDIEndpointSynthesizer
28  *
29  */
30 @interface MIKMIDISynthesizer : NSObject <MIKMIDICommandScheduler>

94 /**
95  * Sets up the AUGraph for the instrument. Do not call this method, as it is
96  * called automatically during initialization.
97  *
98  * The method is provided to give subclasses a chance to override
99  * the AUGraph behavior for the instrument. If you do override it, you will need
100  * to create an AudioUnit instrument and set it to the instrument property. Also,
101  * if you intend to use the graph property, you will be responsible for setting
102  * that as well. DisposeAUGraph() is called on the previous graph when setting
103  * the graph property, and in dealloc.
104  *
105  * @param error If an error occurs, upon returns contains an NSError object that describes the problem. If you are not interested in possible errors, you may pass in NULL.
106  *
107  * @return YES is setting up the graph was succesful, and initialization
108  * should continue, NO if setting up the graph failed and initialization should
109  * return nil.
110  */
111 - (BOOL)setupAUGraphWithError:(NSError **)error;

18 /**
19  * MIKMIDIMappingItem contains information about a mapping between a physical MIDI control,
20  * and a single command supported by a particular MIDI responder object.
21  *
22  * MIKMIDIMappingItem specifies the command type, and MIDI channel for the commands sent by the
23  * mapped physical control along with the control's interaction type (e.g. knob, turntable, button, etc.).
24  * It also specifies the (software) MIDI responder to which incoming commands from the mapped control
25  * should be routed.
26  *
27  */
28 @interface MIKMIDIMappingItem : NSObject <NSCopying>

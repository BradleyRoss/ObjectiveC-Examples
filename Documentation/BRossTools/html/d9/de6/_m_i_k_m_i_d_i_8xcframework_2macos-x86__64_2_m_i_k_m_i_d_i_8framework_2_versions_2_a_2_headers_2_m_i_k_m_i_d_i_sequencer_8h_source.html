22 /**
23  * Types of click track statuses, that determine when the click track will be audible.
24  *
25  * @see clickTrackStatus
26  */
27 typedef NS_ENUM(NSInteger, MIKMIDISequencerClickTrackStatus) {
28  /** The click track will not be heard during playback or recording. */
29  MIKMIDISequencerClickTrackStatusDisabled,
30  /** The click track will only be heard while recording. */
31  MIKMIDISequencerClickTrackStatusEnabledInRecord,
32  /** The click track will only be heard while recording and while the playback position is still in the pre-roll. */
33  MIKMIDISequencerClickTrackStatusEnabledOnlyInPreRoll,
34  /** The click track will always be heard during playback and recording. */
35  MIKMIDISequencerClickTrackStatusAlwaysEnabled

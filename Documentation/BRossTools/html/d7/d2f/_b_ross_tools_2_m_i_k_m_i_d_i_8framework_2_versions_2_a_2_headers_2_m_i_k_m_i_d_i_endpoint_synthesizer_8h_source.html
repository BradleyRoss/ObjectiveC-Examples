19 /**
20  * MIKMIDIEndpointSynthesizer is a subclass of MIKMIDISynthesizer that
21  * provides a very simple way to synthesize MIDI commands coming from a
22  * MIDI endpoint (e.g. from a connected MIDI piano keyboard) to produce sound output.
23  *
24  * To use it, simply create a synthesizer instance with the source you'd like it to play. It will
25  * continue playing incoming MIDI until it is deallocated.
26  *
27  * @see MIKMIDISynthesizer
28  */
29 @interface MIKMIDIEndpointSynthesizer : MIKMIDISynthesizer

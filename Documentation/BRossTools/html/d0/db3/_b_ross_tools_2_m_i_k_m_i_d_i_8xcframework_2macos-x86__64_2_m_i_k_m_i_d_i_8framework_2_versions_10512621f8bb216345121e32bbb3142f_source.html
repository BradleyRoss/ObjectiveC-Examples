14  * Bit-mask constants used to specify MIDI responder types for mapping.
15  * Multiple responder types can be specified by ORing them together.
16  * @see -[MIKMIDIMappableResponder MIDIResponderTypeForCommandIdentifier:]

18 typedef NS_OPTIONS(NSUInteger, MIKMIDIResponderType){
19  /**
20  * Responder does not have a type. Cannot be mapped.
21  */
22  MIKMIDIResponderTypeNone = 0,
23 
24  /**
25  * Type for a MIDI responder that can handle messages from a hardware absolute
26  * knob or slider. That is, one that sends control change messages with an absolute value
27  * depending on its position.
28  */
29  MIKMIDIResponderTypeAbsoluteSliderOrKnob = 1 << 0,
30 
31  /**
32  * Type for a MIDI responder that can handle messages from a hardware relative
33  * knob. That is, a knob that sends a message for each "tick", and whose value
34  * depends on the direction (and possibly velocity) of the knob, rather than its
35  * absolute position.
36  */
37  MIKMIDIResponderTypeRelativeKnob = 1 << 1,
38 
39  /**
40  * Type for a MIDI responder that can handle messages from a hardware turntable-like
41  * jog wheel. These are relative knobs, but typically have *much* higher resolution than
42  * a small relative knob. They may also have a touch/pressure sensitive top to detect when
43  * the user is touching, but not turning the wheel.
44  */
45  MIKMIDIResponderTypeTurntableKnob = 1 << 2,
46 
47  /**
48  * Type for a MIDI responder that can handle messages from a hardware relative knob that
49  * sends messages to simulate an absolute knob. Relative knobs on (at least) Native Instruments
50  * controllers can be configured to send messages like an absolute knob. This can pose the problem
51  * of the knob continuing to turn past its limits (0 and 127) without additional messages being sent.
52  * These knobs can and will be mapped as a regular absolute knob for responders that include MIKMIDIResponderTypeAbsoluteSliderOrKnob
53  * but *not* MIKMIDIResponderTypeRelativeAbsoluteKnob in the type returned by -MIDIResponderTypeForCommandIdentifier:
54  */
55  MIKMIDIResponderTypeRelativeAbsoluteKnob = 1 << 3,
56 
57  /**
58  * Type for a MIDI responder that can handle messages from a hardware button that sends a message when
59  * pressed down, and another message when released.
60  */
61  MIKMIDIResponderTypePressReleaseButton = 1 << 4,
62 
63  /**
64  * Type for a MIDI responder that can handle messages from a hardware button that only sends a single
65  * message when pressed down, without sending a corresponding message upon release.
66  */
67  MIKMIDIResponderTypePressButton = 1 << 5,
68 
69  /**
70  * Convenience type for a responder that can handle messages from any type of knob.
71  */
72  MIKMIDIResponderTypeKnob = (MIKMIDIResponderTypeAbsoluteSliderOrKnob | MIKMIDIResponderTypeRelativeKnob | \
73  MIKMIDIResponderTypeTurntableKnob | MIKMIDIResponderTypeRelativeAbsoluteKnob),
74 
75  /**
76  * Convenience type for a responder that can handle messages from any type of button.
77  */
78  MIKMIDIResponderTypeButton = (MIKMIDIResponderTypePressButton | MIKMIDIResponderTypePressReleaseButton),
79 
80  /**
81  * Convenience type for a responder that can handle messages from any kind of control.
82  */
83  MIKMIDIResponderTypeAll = NSUIntegerMax,

88 /**
89  * This protocol defines methods that that must be implemented by MIDI responder objects to be mapped

91  * during normal operation.

96 /**
97  * The list of identifiers for all commands supported by the receiver.
98  *
99  * A MIDI responder may want to handle incoming MIDI message from more than one control. For example, a view displaying
100  * a list of songs may want to support commands for browsing up and down the list with buttons, or with a knob, as well as a button
101  * to load the selected song. These commands would be for example, KnobBrowse, BrowseUp, BrowseDown, and Load. This way, multiple physical
102  * controls can be mapped to different functions of the same MIDI responder.
103  *
104  * @return An NSArray containing NSString identifers for all MIDI mappable commands supported by the receiver.
105  */
106 - (MIKArrayOf(NSString *) *)commandIdentifiers;
107 
108 /**
109  * The MIDI responder types the receiver will allow to be mapped to the command specified by commandID.
110  *
111  * In the example given for -commandIdentifers, the "KnobBrowse" might be mappable to any physical knob,
112  * while BrowseUp, BrowseDown, and Load are mappable to buttons. The responder would return MIKMIDIResponderTypeKnob
113  * for @"KnobBrowse" while returning MIKMIDIResponderTypeButton for the other commands.
114  *
115  * @param commandID A command identifier string.
116  *
117  * @return A MIKMIDIResponderType bitfield specifing one or more responder type(s).
118  *
119  * @see MIKMIDIResponderType
120  */
121 - (MIKMIDIResponderType)MIDIResponderTypeForCommandIdentifier:(NSString *)commandID; // Optional. If not implemented, MIKMIDIResponderTypeAll will be assumed.
122 
123 @optional
124 
125 /**
126  * Whether the physical control mapped to the commandID in the receiver should
127  * be illuminated, or not.
128  *
129  * Many hardware MIDI devices, e.g. DJ controllers, have buttons that can light
130  * up to show state for the associated function. For example, the play button
131  * could be illuminated when the software is playing. This method allows mapped
132  * MIDI responder objects to communicate the desired state of the physical control
133  * mapped to them.
134  *
135  * Currently MIKMIDI doesn't provide automatic support for actually updating
136  * physical LED status. This must be implemented in application code. For most devices,
137  * this can be accomplished by sending a MIDI message _to_ the device. The MIDI message
138  * should identical to the message that the relevant control sends when pressed, with
139  * a non-zero value to illumniate the control, or zero to turn illumination off.
140  *
141  * @param commandID The commandID for which the associated illumination state is desired.
142  *
143  * @return YES if the associated control should be illuminated, NO otherwise.
144  */
145 - (BOOL)illuminationStateForCommandIdentifier:(NSString *)commandID;

147 @end

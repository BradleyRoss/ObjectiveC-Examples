12 /**
13  * Represents a time signature. Note that in contrast to time signature events in raw MIDI,
14  * the denominator here is the "natural" denominator for the time signature. e.g. 4/4 time
15  * is represented with a numerator of 4 and denominator of 4.
16  */
17 typedef struct {
18  UInt8 numerator; /// The number of beats per measure.
19  UInt8 denominator; // The fraction of a whole note per beat (e.g. 4 here means a quarter note per beat)
20 } MIKMIDITimeSignature;
21 
22 /**
23  * Convenience function to create a MIDITimeSignature struct.
24  *
25  * For example, to create a time signature struct for 4/4 time:
26  * MIKMIDITimeSignatureMake(4, 4)
27  *
28  * @param numerator The numerator for the time signature, or number of beats per measure.
29  * @param denominator The denominator for the time signature, or fraction of a note per beat.
30  *
31  * @return An MIKMIDITimeSignature struct.
32  */
33 NS_INLINE MIKMIDITimeSignature MIKMIDITimeSignatureMake(UInt8 numerator, UInt8 denominator) {
34  MIKMIDITimeSignature ts;
35  ts.numerator = numerator;
36  ts.denominator = denominator;
37  return ts;
38 }

25 /**
26  * MIKMIDISourceEndpoint represents a source (input) MIDI endpoint.
27  * It is essentially an Objective-C wrapper for instances of CoreMIDI's MIDIEndpoint class
28  * which are kMIDIObjectType_Source type endpoints.
29  *
30  * MIDI source endpoints are contained by MIDI entities, which are in turn contained by MIDI devices.
31  * MIDI sources can be connected in order to receive data from them using MIKMIDIDeviceManager's
32  * -connectInput:error:eventHandler:virtualDestinations method.
33  *
34  * Note that MIKMIDISourceEndpoint does not declare any methods of its own. All its methods can be
35  * found on its superclasses: MIKMIDIEndpoint and MIKMIDIObject.
36  *
37  * @see -[MIKMIDIDeviceManager connectInput:error:eventHandler:virtualDestinations]
38  * @see -[MIKMIDIDeviceManager disconnectInput:]
39  */

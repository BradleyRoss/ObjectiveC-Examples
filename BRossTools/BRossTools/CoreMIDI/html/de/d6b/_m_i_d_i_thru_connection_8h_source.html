24 /*!
25  @header MIDIThruConnection.h
26 
27  This header defines functions to create MIDI play-through connections
28  between the MIDI sources and destinations. These connections may be
29  persistent or transitory, owned by a client.
30 
31  By using connections instead of doing MIDI Thru operations themselves,
32  the overhead of moving MIDI messages between the server and the client
33  for thru-ing is reduced.
34 
35  The aim of these functions is to permit as flexible a set of transformations
36  as possible while keeping the API and data structures relatively simple.
37 */
38 // -----------------------------------------------------------------------------

132 /*!
133  @struct MIDIControlTransform
134  @abstract Describes a transformation of MIDI control change events.
135 
136  A single MIDIThruConnectionParams may describe any number of transformations to control
137  events. It is important that multiple transformations are ordered correctly: filter out,
138  remap, then alter values.
139 
140  All transformations are done internally using 14-bit values, so for example, when doing
141  an add/min/max transform on a 7-bit control value, the parameter must be a 14-bit value.
142  For example, to add 10 to a control value, param must be (10 << 7) = 1280.
143 
144  As per the MIDI specification, a number of controls are interpreted specially:
145 
146  Control | Function
147  --------|---------
148  32-63 | the LSBs of 0-31
149  6/38 | data entry
150  96, 97 | data increment, decrement
151  98-101 | NRPN/RPN
152 
153  @field controlType The type of control specified by controlNumber
154  @field remappedControlType If transform is kMIDITransform_MapControl, the output control type
155  @field controlNumber The control number to be affected.
156  @field transform The type of transformation to be applied to the event values.
157  @field param An argument to the transformation method (see description of MIDITransformType).
158 */
159 struct MIDIControlTransform {
160  MIDITransformControlType controlType;
161  MIDITransformControlType remappedControlType;
162  UInt16 controlNumber;
163  MIDITransformType transform;
164  SInt16 param;
165 };

187 /*!
188  @struct MIDIThruConnectionParams
189  @abstract Describes a set of MIDI routings and transformations.
190 
191  The remainder of the structure is variably-sized. It contains numControlTransform instances of
192  MIDIControlTransform, followed by numMaps instances of MIDIValueMap.
193 
194  @field version Version of this structure; must be 0.
195  @field numSources The number of valid sources in the following array.
196  @field sources All MIDI generated by these sources is routed into this connection for processing
197  and distribution to destinations.
198  @field numDestinations The number of valid destinations in the following array.
199  @field destinations All MIDI output from the connection is routed to these destinations.
200  @field channelMap Maps each of the source 16 MIDI channels to channel 0-15 (1-16) or 0xFF when
201  MIDI from a channel is to be filtered out.
202  @field lowVelocity Note events with a velocity less than this value are filtered out.
203  @field highVelocity Note events with a velocity greater than this, if it is not 0, are filtered out.
204  @field lowNote See highNote.
205  @field highNote If highNote >= lowNote, then notes outside this range are filtered out.
206  If lowNote > highNote, then notes <i>inside</i> this range are filtered out.
207  This applies to note and polyphonic key pressure events.
208  These fields are ignored if a there is a MIDIValueMap applying to noteNumber.
209  @field noteNumber Specifies how MIDI note numbers are transformed.
210  @field velocity Specifies how MIDI note velocities are transformed.
211  @field keyPressure Specifies how MIDI polyphonic key pressure events are transformed.
212  @field channelPressure Specifies how MIDI monophonic (channel) pressure events are transformed.
213  @field programChange Specifies how MIDI program change events are transformed.
214  @field pitchBend Specifies how MIDI pitch bend events are transformed.
215  @field filterOutSysEx If 1, specifies that system-exclusive messages are to be filtered out.
216  @field filterOutMTC If 1, specifies that MIDI Time Code messages are to be filtered out.
217  @field filterOutBeatClock If 1, specifies the MIDI clock, play, stop, and resume messages are to
218  be filtered out.
219  @field filterOutTuneRequest If 1, specifies that MIDI Tune Request messages are to be filtered out.
220  @field reserved2 Must be 0.
221  @field filterOutAllControls If 1, specifies that all MIDI continuous control messages are to be filtered out.
222  @field numControlTransforms The number of control transformations in the variable-length portion of the struct.
223  @field numMaps The number of MIDIValueMaps in the variable-length portion of the struct.
224  @field reserved3 Must be 0.
225 */

230  UInt32 numDestinations;

232 
233  UInt8 channelMap[16];
234  UInt8 lowVelocity, highVelocity;
235  UInt8 lowNote, highNote;
236  MIDITransform noteNumber;
237  MIDITransform velocity;
238  MIDITransform keyPressure;
239  MIDITransform channelPressure;
240  MIDITransform programChange;
241  MIDITransform pitchBend;
242 
243  UInt8 filterOutSysEx;
244  UInt8 filterOutMTC;
245  UInt8 filterOutBeatClock;
246  UInt8 filterOutTuneRequest;
247  UInt8 reserved2[3];
248  UInt8 filterOutAllControls;
249 
250  UInt16 numControlTransforms;
251  UInt16 numMaps;
252  UInt16 reserved3[4];

254  // remainder of structure is variable-length:
255  // MIDIControlTransform controls[];
256  // MIDIValueMap maps[];
